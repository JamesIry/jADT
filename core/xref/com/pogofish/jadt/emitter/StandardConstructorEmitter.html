16  package com.pogofish.jadt.emitter;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  import java.util.logging.Logger;
21  
22  import com.pogofish.jadt.ast.Constructor;
23  import com.pogofish.jadt.sink.Sink;
24  
25  
26  public class StandardConstructorEmitter implements ConstructorEmitter {
27  	private static final Logger logger = Logger.getLogger(StandardConstructorEmitter.class.toString());
28      private final ClassBodyEmitter classBodyEmitter;
29      
30      public StandardConstructorEmitter(ClassBodyEmitter classBodyEmitter) {
31          super();
32          this.classBodyEmitter = classBodyEmitter;
33      }
34  
35      @Override
36      public void constructorFactory(Sink sink, String dataTypeName, List<String> typeParameters, Constructor constructor) {
37          classBodyEmitter.constructorFactory(sink, dataTypeName, constructor.name, typeParameters, constructor);
38      }
39      
40      @Override
41      public void constructorDeclaration(Sink sink, Constructor constructor, String dataTypeName, List<String> typeParameters) {
42      	logger.finer("Generating constructor class for " + constructor.name + " in datatype " + dataTypeName);
43      	sink.write("   public static final class " + constructor.name);
44          classBodyEmitter.emitParameterizedTypeName(sink, typeParameters);
45          sink.write(" extends " + dataTypeName);
46          classBodyEmitter.emitParameterizedTypeName(sink, typeParameters);
47          sink.write(" {\n");
48          
49          classBodyEmitter.emitConstructorMethod(sink, constructor);
50          sink.write("\n\n");
51          
52          emitAccept(sink, typeParameters);
53          sink.write("\n\n");
54          
55          classBodyEmitter.emitHashCode(sink, constructor);
56          sink.write("\n\n");
57          
58          classBodyEmitter.emitEquals(sink, constructor, typeParameters);
59          sink.write("\n\n");
60          
61          classBodyEmitter.emitToString(sink, constructor);
62          sink.write("\n\n");
63          
64          sink.write("   }");
65      }
66      
67      private void emitAccept(Sink sink, List<String> typeArguments) {
68          final List<String> visitorTypeArguments = new ArrayList<String>(typeArguments);
69          visitorTypeArguments.add("ResultType");
70          
71          sink.write("      @Override\n");
72          sink.write("      public <ResultType> ResultType match(MatchBlock");
73          classBodyEmitter.emitParameterizedTypeName(sink, visitorTypeArguments);
74          sink.write(" matchBlock) { return matchBlock._case(this); }\n");
75          sink.write("\n");
76          sink.write("      @Override\n");
77          sink.write("      public void _switch(SwitchBlock");
78          classBodyEmitter.emitParameterizedTypeName(sink, typeArguments);
79          sink.write(" switchBlock) { switchBlock._case(this); }");
80      }

83  }
