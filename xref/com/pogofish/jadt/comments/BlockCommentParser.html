26  /**
27   * Parser for BlockComment.
28   * 
29   * @author jiry
30   */
31  public class BlockCommentParser {
32      public JavaComment parse(final Reader BlockComment) {
33          return Util.execute(new ExceptionAction<JavaComment>() {
34  
35              @Override
36              public JavaComment doAction() throws Throwable {
37  
38                  BlockCommentParserImpl impl = new BlockCommentParserImpl(
39                          new JavaCCReader(BlockComment));
40                  return impl.blockComment();
41              }
42          });
43      }
44  }
