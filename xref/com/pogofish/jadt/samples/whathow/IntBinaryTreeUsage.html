44      /**
45       * Find the max Integer of a tree where null means empty tree
46       */
47      // START SNIPPET: max  
48      public Integer max(IntBinaryTree tree)  {
49          return tree.match(new IntBinaryTree.MatchBlock<Integer>() {
50             @Override
51             public Integer _case(Node x) {
52                final Integer maxLeft = max(x.left);
53                final int l = maxLeft == null ? Integer.MIN_VALUE : maxLeft;
54  
55                final Integer maxRight = max(x.right);
56                final int r = maxRight == null ? Integer.MIN_VALUE : maxRight;
57  
58                return Math.max(Math.max(l, r), x.value);
59             }
60  
61             @Override
62             public Integer _case(EmptyTree x) {
63                return null;
64             }
65          });
66       }
67      // END SNIPPET: max
