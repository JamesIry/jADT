290     {

294 
295       for (;;) // Read all the backslashes
296       {
297         if (++bufpos == available)
298           AdjustBuffSize();
299 
300         try
301         {
302           if ((buffer[bufpos] = c = ReadByte()) != '\\')
303           {
304             UpdateLineColumn(c);
305             // found a non-backslash char.
306             if ((c == 'u') && ((backSlashCnt & 1) == 1))
307             {
308               if (--bufpos < 0)
309                 bufpos = bufsize - 1;
310 
311               break;
312             }
313 
314             backup(backSlashCnt);
315             return '\\';
316           }
317         }
318         catch(java.io.IOException e)
319         {
320 	  // We are returning one backslash so we should only backup (count-1)
321           if (backSlashCnt > 1)
322             backup(backSlashCnt-1);
323 
324           return '\\';
325         }
326 
327         UpdateLineColumn(c);
328         backSlashCnt++;
329       }
330 
331       // Here, we have seen an odd number of backslash's followed by a 'u'
332       try
333       {
334         while ((c = ReadByte()) == 'u')
335           ++column;
336 
337         buffer[bufpos] = c = (char)(hexval(c) << 12 |
338                                     hexval(ReadByte()) << 8 |
339                                     hexval(ReadByte()) << 4 |
340                                     hexval(ReadByte()));

343       }
344       catch(java.io.IOException e)
345       {
346         throw new Error("Invalid escape character at line " + line +
347                                          " column " + column + ".");
348       }
349 
350       if (backSlashCnt == 1)
351         return c;
352       else
353       {
354         backup(backSlashCnt - 1);

356       }
357     }
358     else

360       UpdateLineColumn(c);

588   {

592     if (bufpos >= tokenBegin)
593     {
594       len = bufpos - tokenBegin + inBuf + 1;

598       len = bufsize - tokenBegin + bufpos + 1 + inBuf;
599     }

601     int i = 0, j = 0, k = 0;
602     int nextColDiff = 0, columnDiff = 0;
603 
604     while (i < len && bufline[j = start % bufsize] == bufline[k = ++start % bufsize])
605     {
606       bufline[j] = newLine;
607       nextColDiff = columnDiff + bufcolumn[k] - bufcolumn[j];
608       bufcolumn[j] = newCol + columnDiff;
609       columnDiff = nextColDiff;
610       i++;
611     }
612 
613     if (i < len)
614     {
615       bufline[j] = newLine++;
616       bufcolumn[j] = newCol + columnDiff;
617 
618       while (i++ < len)
619       {
620         if (bufline[j = start % bufsize] != bufline[++start % bufsize])
621           bufline[j] = newLine++;
622         else
