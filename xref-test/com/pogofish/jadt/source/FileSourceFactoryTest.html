35  /**
36   * Make sure FileSourceFactory does its job properly

40  public class FileSourceFactoryTest {
41      /**
42       * A comparator that gives a predictable sort on sources based on their name
43       * 
44       * @author jiry
45       */
46      private static final class SourceComparator implements Comparator<Source> {
47          @Override
48          public int compare(Source source1, Source source2) {
49              return source1.getSrcInfo().compareTo(source2.getSrcInfo());
50          }
51      }

53      /**
54       * On a valid file should get the data from the file.  This test must write to the file system
55       * @throws IOException
56       */
57      @Test
58      public void testValidFile() throws IOException {
59          final File temp = File.createTempFile("testFactory", "java");
60          try {
61              final BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(temp), "UTF-8"));
62              try {
63                  writer.write("hello");
64              } finally {
65                  writer.close();
66              }
67              
68              final SourceFactory factory = new FileSourceFactory();
69              final List<? extends Source> sources = factory.createSources(temp.getAbsolutePath());
70              assertEquals(1, sources.size());
71              final Source source = sources.get(0);
72              final BufferedReader reader = source.createReader();
73              try {
74                  assertEquals("hello", reader.readLine());
75                  assertEquals(temp.getAbsolutePath(), source.getSrcInfo());
76              } finally {
77                  reader.close();
78              }
79          } finally {
80              temp.delete();
81          }
82      }

109     /**
110      * Test to see if FileSourceFactoryTest works with a directory
111      */
112     @Test
113     public void testDir() throws IOException {
114         final File tempDir = TestUtil.createTmpDir();
115         try {
116             final List<File> tempFiles = new ArrayList<File>();
117             try {
118                 for (int i=0; i<5; i++) {
119                     tempFiles.add(createTempFile(tempDir, i));
120                 }
121                 
122                 final SourceFactory factory = new FileSourceFactory();
123                 final List<? extends Source> sources = factory.createSources(tempDir.getAbsolutePath());
124                 assertEquals(5, sources.size());
125                 
126                 // sort them so that we look at them in 0...4 order
127                 Collections.sort(sources, new SourceComparator());
128                 
129                 for (int i=0; i<5; i++) {
130                     checkSource(i, tempDir, sources.get(i));
131                 }
132                
133             } finally {
134                 for (File tempFile : tempFiles) {
135                     tempFile.delete();
136                 }
137             }
138         } finally {
139             tempDir.delete();
140         }      
141     }

143     private void checkSource(int i, File parent, Source source) throws IOException {
144         final BufferedReader reader = source.createReader();
145         try {
146             assertEquals(parent.getAbsolutePath() + "/" + i + ".jadt", source.getSrcInfo());
147             final String line = reader.readLine();
148             assertEquals("hello" + i, line);
149         } finally {
150             reader.close();
151         }        
152     }
153 
154     private File createTempFile(File parentDir, int i) {
155         final File tempFile = new File(parentDir, i + ".jadt");
156         try {
157             final BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(tempFile), "UTF-8"));
158             try {
159                 writer.write("hello" + i);
160             } finally {
161                 writer.close();
162             }
163             return tempFile;
164         } catch (Throwable t) {
165             tempFile.delete();
166             throw new RuntimeException(t);
167         }
168     }
