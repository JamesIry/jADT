30  public class GenericBinaryTreeUsage {
31      /**
32       * Creates an example IntBinaryTree with some fixed values
33       */
34      public BinaryTree<String> createExample() {
35          // START SNIPPET: sampleTree
36          BinaryTree<String> empty = BinaryTree.<String>_EmptyTree();
37  
38          return _Node("hello", _Node("goodbye", empty, empty), _Node("whatever", empty, _Node("foo", empty, empty)));        
39          // END SNIPPET: sampleTree
40      } 
41      
42      /**
43       * Find the string that sorts highest in a tree where null 
44       * is considered smallest and otherwise strings are compared with compareTo
45       */
46      // START SNIPPET: max
47      public String max(BinaryTree<String> tree)  {
48          return tree.accept(new BinaryTree.Visitor<String, String>() {
49             @Override
50             public String visit(Node<String> x) {
51                 final String maxLeft = max(x.left);
52                 final String maxRight = max(x.right);
53  
54                 return maxString(maxString(maxLeft, maxRight), x.value);
55             }
56  
57             @Override
58             public String visit(EmptyTree<String> x) {
59                return null;
60             }
61          });
62       }
63      
64      private String maxString(String l, String r) {
65          return l == null ? r : (r == null ? l : (l.compareTo(r) >= 0 ? l : r));
66      }
67      // END SNIPPET: max
68  }
