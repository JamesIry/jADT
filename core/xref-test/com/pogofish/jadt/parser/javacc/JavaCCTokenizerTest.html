62       * Create a tokenizer that will read from the given string
63       */
64      private BaseJavaCCParserImplTokenManager tokenizer(String testString) {
65          final Source source = new StringSource("TokenizerTest", testString);
66          return new BaseJavaCCParserImplTokenManager(new JavaCharStream(
67                  source.createReader()));

293     /**
294      * Check that the next tokenType, symbol, and lineNo from the tokenizer are
295      * as expected
296      */
297     private void check(BaseJavaCCParserImplTokenManager tokenizer,
298             String expectedSymbol, int expectedTokenType, int expectedLineNo) {
299         final Token token = tokenizer.getNextToken();
300         final String actualSymbol = token.kind == EOF ? "<EOF>" : token.image;
301         assertEquals("Expected token type " + expectedTokenType
302                 + " with symbol " + expectedSymbol + " but got " + token.kind
303                 + " with symbol " + actualSymbol, expectedTokenType, token.kind);
304         assertEquals("Got correct token type " + expectedTokenType
305                 + " but got wrong symbol", expectedSymbol, actualSymbol);
306         assertEquals("Got correct token type and symbol but wrong line number",
307                 expectedLineNo, token.beginLine);
308     }
