45      @Test
46      public void testStripTags() {
47          testStripTags("/** */\n", list("/** */"), set("@foo"));
48          testStripTags("/** @bar whatever */\n", list("/** @bar whatever */"), set("@foo"));
49          testStripTags("/** */\n", list("/** @foo whatever */"), set("@foo"));
50          testStripTags("/** blah blah @hello there\n */\n", list("/** blah blah @hello there\n * @foo first \n @foo second *\n*/"), set("@foo"));
51          
52          // make sure strip tags leaves other comment types alone
53          final CommentProcessor commentProcessor = new CommentProcessor();
54          @SuppressWarnings("unchecked")
55          List<JavaComment> comments = list(_JavaEOLComment("//whatever"), _JavaBlockComment(list(list(_BlockWord("/*"), _BlockWhiteSpace(" "), _BlockWord("*/")))));
56          assertEquals(comments, commentProcessor.stripTags(set("@foo"), comments));
57      }

59      private void testStripTags(String expected, List<String> inputs, Set<String> tags) {
60          final JavaDocParser parser = new JavaDocParser();
61          final List<JavaComment> outputs = new ArrayList<JavaComment>(inputs.size());
62          final CommentProcessor commentProcessor = new CommentProcessor();
63          for (String input : inputs) {
64              final JavaComment comment = parser.parse(new StringReader(input));
65              outputs.add(comment);
66          }
67          final List<JavaComment> stripped = commentProcessor.stripTags(tags, outputs);
68          final String actual = ASTPrinter.printComments("", stripped);
69          assertEquals(expected, actual);    
70     }
