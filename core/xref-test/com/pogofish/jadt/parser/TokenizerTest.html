38      /**
39       * Test that an IOException in the tokenizer is converted to a RuntimeException
40       */
41      @Test
42      public void testIOException() {
43          // a dummy reader that throws an exception on any action
44          final Reader reader = new Reader() {
45              @Override
46              public int read(char[] cbuf, int off, int len) throws IOException {
47                  throw new IOException("TestException");
48              }
49  
50              @Override
51              public void close() throws IOException {
52                  throw new IOException("TestException");
53              }
54          };
55          
56          // a dummy source that uses that reader
57          final Source source = new Source() {            
58              @Override
59              public String getSrcInfo() {
60                  return "TokenizerTest";
61              }
62              
63              @Override
64              public Reader getReader() {
65                  return reader;
66              }
67              
68              @Override
69              public void close() {
70                  try {
71                      reader.close();
72                  } catch (IOException e) {
73                      throw new RuntimeException(e);
74                  }
75              }
76          };
77          
78          final Tokenizer tokenizer = new Tokenizer(source);
79  
80          try {
81              // should throw
82              tokenizer.peek();
83              fail("Did not get an exception from tokenizer");
84          } catch (RuntimeException e) {
85              assertTrue("Got wrong exception " + e, e.getCause() instanceof IOException);
86          }
87      }

99      @Test
100     public void testComments() {
101         final Tokenizer tokenizer = tokenizer("/*\nCopyright*/hello//comment\nworld/*another comment*/oh");
102         check(tokenizer, "hello", TokenType.IDENTIFIER, 2);
103         check(tokenizer, "world", TokenType.IDENTIFIER, 3);
104         check(tokenizer, "oh", TokenType.IDENTIFIER, 3);
105         check(tokenizer, "<EOF>", TokenType.EOF, 3);

148     @Test
149     public void testBadIdentifiers() {
150         final Tokenizer tokenizer = tokenizer("hello. 42 ?");
151         // bad because of the trailing dot
152         check(tokenizer, "hello.", TokenType.UNKNOWN, 1);
153         // bad because identifiers can't start with numbers (jADT doesn't care about numbers)
154         check(tokenizer, "42", TokenType.UNKNOWN, 1);
155         // just bad
156         check(tokenizer, "?", TokenType.UNKNOWN, 1);

165         check(tokenizer, "<", TokenType.LANGLE, 1);
166         check(tokenizer, ">", TokenType.RANGLE, 1);

168         check(tokenizer, "(", TokenType.LPAREN, 1);
169         check(tokenizer, ")", TokenType.RPAREN, 1);
170         check(tokenizer, ",", TokenType.COMMA, 1);
171         check(tokenizer, "[", TokenType.LBRACKET, 1);
172         check(tokenizer, "]", TokenType.RBRACKET, 1);
173         check(tokenizer, "|", TokenType.BAR, 1);
174         // this one is tested to provide coverage of a default case in the Tokenizer
175         check(tokenizer, "*", TokenType.UNKNOWN, 1);
176         check(tokenizer, "<EOF>", TokenType.EOF, 1);

247     /**
248      * Check that the next tokenType, symbol, and lineNo from the tokenizer are as expected
249      */
250     private void check(Tokenizer tokenizer, String symbol, TokenType tokenType, int lineNo) {
251         assertTrue("Expected token type " + tokenType + " with symbol " + symbol + " but got " + tokenizer.peek()
252                 + " with symbol " + tokenizer.lastSymbol(), tokenizer.accept(tokenType));
253         assertEquals("Got correct token type " + tokenType + " but got wrong symbol", symbol, tokenizer.lastSymbol());
254         assertEquals("Got correct token type and symbol but wrong line number", lineNo, tokenizer.lineno());
255     }
