52       * Turns a list of JavaComment into list of comments that only has javadoc comments
53       */
54      public List<JavaComment> javaDocOnly(final List<JavaComment> originals) {
55          final List<JavaComment> results = new ArrayList<JavaComment>(
56                  originals.size());
57          for (JavaComment comment : originals) {
58              comment._switch(new JavaComment.SwitchBlockWithDefault() {
59                  
60                  @Override
61                  public void _case(JavaDocComment x) {
62                      results.add(x);
63                  }
64  
65                  @Override
66                  protected void _default(JavaComment x) {
67                      // do nothing, we only want to add javadocs
68                  }
69              });
70          }
71          return results;
72      }

74      /**
75       * Produce a copy of a list of comments where all javadoc comments have had
76       * the specified tags removed. Non javadoc comments are left alone.
77       * 
78       * @param tagNames
79       *            Set of tag names (including leading @) to be removed
80       * @param originals
81       *            List of Comments to be processed
82       * @return a new list of comments with the specified tags removed
83       */
84      public List<JavaComment> stripTags(final Set<String> tagNames,
85              List<JavaComment> originals) {
86          final List<JavaComment> results = new ArrayList<JavaComment>(
87                  originals.size());
88          for (JavaComment original : originals) {
89              results.add(original
90                      .match(new JavaComment.MatchBlock<JavaComment>() {
91  
92                          @Override
93                          public JavaComment _case(JavaDocComment x) {
94                              final List<JDTagSection> newTagSections = new ArrayList<JDTagSection>(
95                                      x.tagSections.size());
96                              for (JDTagSection tagSection : x.tagSections) {
97                                  if (!tagNames.contains(tagSection.name)) {
98                                      newTagSections.add(tagSection);
99                                  }
100                             }
101                             return _JavaDocComment(x.start, x.generalSection,
102                                     newTagSections, x.end);
103                         }
104 
105                         @Override
106                         public JavaComment _case(JavaBlockComment x) {
107                             return x;
108                         }
109 
110                         @Override
111                         public JavaComment _case(JavaEOLComment x) {
112                             return x;
113                         }
114                     }));
115         }
116         return results;
117     }

119     /**
120      * Produce a copy of a list of JavaDoc comments by pulling specified parameter tags out of an orignal list of comments.  
121      * Block and EOL comments are skipped.  JavaDoc comments are stripped down to only have the content of the specified 
122      * param tag
123      */
124     public List<JavaComment> paramDoc(final String paramName, List<JavaComment> originals) {
125         final List<JavaComment> results = new ArrayList<JavaComment>(originals.size());
126         for (JavaComment original : originals) {
127             original._switch(new JavaComment.SwitchBlock() {
128                 @Override
129                 public void _case(JavaDocComment x) {
130                     paramDocSections(paramName, x.tagSections, results);
131                 }
132 
133                 @Override
134                 public void _case(JavaBlockComment x) {
135                     // skip
136                 }
137 
138                 @Override
139                 public void _case(JavaEOLComment x) {
140                     // skip
141                 }
142             });
143         }
144         return results;
145     }

148     private void paramDocSections(String paramName,
149             List<JDTagSection> tagSections, List<JavaComment> results) {
150         for (JDTagSection section : tagSections) {
151             if (section.name.equals("@param")) {
152                 paramDocSection(paramName, section.tokens, results);
153             }
154         }
155     }

157     private static enum ParamState {
158         BEGIN, ASTERISK, TAGGED, ACCUMULATING, DEAD;
159     }
160 
161     private void paramDocSection(final String paramName, final List<JDToken> tokens,
162             List<JavaComment> results) {
163         final ParamState state[] = new ParamState[]{ParamState.BEGIN};
164         @SuppressWarnings("unchecked")
165         final List<JDToken>[] accum = new List[1];
166         
167         for (JDToken token : tokens) {
168             token._switch(new JDToken.SwitchBlock() {
169                 
170                 @Override
171                 public void _case(JDWhiteSpace x) {
172                     if (state[0] == ParamState.ACCUMULATING) {
173                         accum[0].add(x);
174                     }                    
175                 }
176                 
177                 @Override
178                 public void _case(JDWord x) {
179                     switch(state[0]) {
180                     case ACCUMULATING:
181                         accum[0].add(x);
182                         break;
183                     case TAGGED: 
184                         if (x.word.equals(paramName)) {
185                             state[0] = ParamState.ACCUMULATING;
186                             accum[0] = new ArrayList<JDToken>(tokens.size() - 1);
187                             accum[0].add(_JDEOL("\n"));
188                             accum[0].add(_JDWhiteSpace(" "));
189                             accum[0].add(_JDAsterisk());
190                         } else {
191                             state[0] = ParamState.DEAD;
192                         }
193                         break;
194                     default:
195                         state[0] = ParamState.DEAD;
196                         break;
197                     }                 
198                 }
199                 
200                 @Override
201                 public void _case(JDTag x) {
202                     switch(state[0]) {
203                     case ACCUMULATING:
204                         accum[0].add(x);
205                         break;
206                     case BEGIN:
207                         state[0] = ParamState.TAGGED;
208                         break;
209                     case ASTERISK:
210                         state[0] = ParamState.TAGGED;
211                         break;
212                     default:
213                         state[0] = ParamState.DEAD;
214                         break;
215                     }    
216                 }
217                 
218                 @Override
219                 public void _case(JDEOL x) {
220                     if (state[0] == ParamState.ACCUMULATING) {
221                         accum[0].add(x);
222                     }                    
223                 }
224                 
225                 @Override
226                 public void _case(JDAsterisk x) {
227                     switch(state[0]) {
228                     case ACCUMULATING:
229                         accum[0].add(x);
230                         break;
231                     case BEGIN:
232                         state[0] = ParamState.ASTERISK;
233                         break;
234                     default:
235                         state[0] = ParamState.DEAD;
236                         break;
237                     }  
238                 }
239             });
240         }
241         
242         if(state[0] == ParamState.ACCUMULATING) {
243             results.add(_JavaDocComment("/**", accum[0], Util.<JDTagSection>list(), "*/"));
244         }
245         
246     }

248     /**
249      * Align a list of comments on the left marging.  EOLComments are left alone.  Block and JavaDoc comments are aligned by making every line that starts
250      * with a * be one space in from the comment opener.  Lines that don't start with * are left alone.
251      */
252     public List<JavaComment> leftAlign(List<JavaComment> originals) {
253         final List<JavaComment> results = new ArrayList<JavaComment>(
254                 originals.size());
255         for (JavaComment original : originals) {
256             results.add(original
257                     .match(new JavaComment.MatchBlock<JavaComment>() {
258 
259                         @Override
260                         public JavaComment _case(JavaDocComment x) {
261                             final List<JDToken> leadingWhiteSpace = new ArrayList<JDToken>(1);
262                             final LeftAlignState state[] = new LeftAlignState[] { LeftAlignState.IN_LINE };
263                             return _JavaDocComment(x.start, leftAlignSection(x.generalSection, x.tagSections.isEmpty(), leadingWhiteSpace, state), leftAlignSections(x.tagSections, leadingWhiteSpace, state), x.end);
264                         }
265 
266 
267                         @Override
268                         public JavaComment _case(JavaBlockComment x) {
269                             return _JavaBlockComment(leftAlignBlock(x.lines));
270                         }
271 
272                         @Override
273                         public JavaComment _case(JavaEOLComment x) {
274                             return x;
275                         }
276                     }));
277         }
278         return results;        
279     }

282     private List<JDTagSection> leftAlignSections(
283             List<JDTagSection> tagSections, List<JDToken> leadingWhiteSpace, LeftAlignState[] state) {
284         final int size = tagSections.size();
285         final List<JDTagSection> newSections = new ArrayList<JDTagSection>(size);
286         int count = 0;
287         for (JDTagSection section : tagSections) {
288             count ++;
289             newSections.add(_JDTagSection(section.name, leftAlignSection(section.tokens, count == size, leadingWhiteSpace, state)));
290         }
291         return newSections;
292     }

294     private List<JDToken> leftAlignSection(List<JDToken> originalSection, boolean lastSection, final List<JDToken> leadingWhiteSpace, final LeftAlignState[] state) {
295         final List<JDToken> result = new ArrayList<JDToken>(
296                 originalSection.size() + 1);
297 
298         for (JDToken token : originalSection) {
299             token._switch(new JDToken.SwitchBlockWithDefault() {
300                 @Override
301                 public void _case(JDWhiteSpace x) {
302                     switch (state[0]) {
303                     case START_LINE:
304                         leadingWhiteSpace.add(x);
305                         break;
306                     case IN_LINE:
307                         result.add(x);
308                         break;
309                     }
310                 }
311 
312                 @Override
313                 public void _case(JDEOL x) {
314                     switch (state[0]) {
315                     case START_LINE:
316                         result.addAll(leadingWhiteSpace);
317                         leadingWhiteSpace.clear();
318                         result.add(x);
319                         break;
320                     case IN_LINE:
321                         result.add(x);
322                         break;
323                     }
324                     state[0] = LeftAlignState.START_LINE;
325                 }
326 
327                 @Override
328                 public void _case(JDAsterisk x) {
329                     switch (state[0]) {
330                     case START_LINE:
331                         result.add(_JDWhiteSpace(" "));
332                         leadingWhiteSpace.clear();
333                         result.add(x);
334                         state[0] = LeftAlignState.IN_LINE;
335                         break;
336                     case IN_LINE:
337                         result.add(x);
338                         break;
339                     }
340                 }
341 
342                 @Override
343                 protected void _default(JDToken x) {
344                     switch (state[0]) {
345                     case START_LINE:
346                         result.addAll(leadingWhiteSpace);
347                         leadingWhiteSpace.clear();
348                         result.add(x);
349                         state[0] = LeftAlignState.IN_LINE;
350                         break;
351                     case IN_LINE:
352                         result.add(x);
353                         break;
354                     }
355                 }
356             });
357         }
358         
359         if (lastSection && state[0] == LeftAlignState.START_LINE) {
360            result.add(_JDWhiteSpace(" "));         
361         }
362 
363         return result;
364     }

371     private List<List<BlockToken>> leftAlignBlock(
372             List<List<BlockToken>> lines) {
373         final List<List<BlockToken>> results = new ArrayList<List<BlockToken>>(lines.size());
374         for (List<BlockToken> line : lines) {
375             results.add(leftAlignLign(line));
376         }
377         return results;
378     }

380     private List<BlockToken> leftAlignLign(List<BlockToken> line) {
381         final List<BlockToken> result = new ArrayList<BlockToken>(line.size() + 1);
382         final List<BlockToken> leadingWhiteSpace = new ArrayList<BlockToken>(1);
383         final LeftAlignState state[] = new LeftAlignState[] { LeftAlignState.START_LINE };
384         
385         for(BlockToken token: line) {
386             token._switch(new BlockToken.SwitchBlock() {
387                 
388                 @Override
389                 public void _case(BlockEOL x) {
390                     switch (state[0]) {
391                     case START_LINE:
392                         result.addAll(leadingWhiteSpace);
393                         leadingWhiteSpace.clear();
394                         result.add(x);
395                         break;
396                     case IN_LINE:
397                         result.add(x);
398                         break;
399                     }
400                     state[0] = LeftAlignState.START_LINE;
401                 }
402                 
403                 @Override
404                 public void _case(BlockWhiteSpace x) {
405                     switch (state[0]) {
406                     case START_LINE:
407                         leadingWhiteSpace.add(x);
408                         break;
409                     case IN_LINE:
410                         result.add(x);
411                         break;
412                     }
413                 }
414                 
415                 @Override
416                 public void _case(BlockWord x) {
417                     if (x.word.startsWith("*")) {
418                         switch (state[0]) {
419                         case START_LINE:
420                             result.add(_BlockWhiteSpace(" "));
421                             leadingWhiteSpace.clear();
422                             result.add(x);
423                             state[0] = LeftAlignState.IN_LINE;
424                             break;
425                         case IN_LINE:
426                             result.add(x);
427                             break;
428                         }
429                     } else {
430                         switch (state[0]) {
431                         case START_LINE:
432                             result.addAll(leadingWhiteSpace);
433                             leadingWhiteSpace.clear();
434                             result.add(x);
435                             state[0] = LeftAlignState.IN_LINE;
436                             break;
437                         case IN_LINE:
438                             result.add(x);
439                             break;
440                         }                        
441                     }
442                 }
443             });
444         }
445         
446         return result;
447     }
