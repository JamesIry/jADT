29      /* (non-Javadoc)
30       * @see pogofish.jadt.emitter.ClassBodyEmitter#constructorFactory(pogofish.jadt.emitter.Target, java.lang.String, java.lang.String, pogofish.jadt.ast.Constructor)
31       */
32      @Override
33      public void constructorFactory(Target target, String dataTypeName, String factoryName, Constructor constructor) {
34          if (constructor.args.isEmpty()) {
35              target.write("   public static final " + dataTypeName + " _" + factoryName + " = new " + constructor.name + "();");
36          } else {
37              target.write("   public static final " + dataTypeName + " _" + factoryName + "(");
38              constructorArgs(target, constructor, true);
39              target.write(") { return new " + constructor.name + "("); 
40              constructorArgs(target, constructor, false);
41              target.write("); }");            
42          }
43      }
44      
45      private void constructorArgs(Target target, Constructor constructor, boolean withTypes) {
46          boolean first = true;
47          for (Arg arg : constructor.args) {
48              if (first) {
49                  first = false;
50              } else {
51                  target.write(", ");
52              }
53              target.write(constructorArg(arg, withTypes));
54          }
55      }

58      /* (non-Javadoc)
59       * @see pogofish.jadt.emitter.ClassBodyEmitter#emitConstructorMethod(pogofish.jadt.emitter.Target, pogofish.jadt.ast.Constructor)
60       */
61      @Override
62      public void emitConstructorMethod(Target target, Constructor constructor) {
63          for (Arg arg : constructor.args) {
64              target.write("      public final " + Printer.print(arg.type) + " " + arg.name + ";");
65              target.write("\n");
66          }
67          target.write("\n      public " + constructor.name + "("); 
68          constructorArgs(target, constructor, true);        
69          target.write(") {");
70          for (Arg arg : constructor.args) {
71              target.write("\n         this." + arg.name + " = " + arg.name + ";");
72          }
73          target.write("\n      }");
74      }

76      /* (non-Javadoc)
77       * @see pogofish.jadt.emitter.ClassBodyEmitter#emitToString(pogofish.jadt.emitter.Target, pogofish.jadt.ast.Constructor)
78       */
79      @Override
80      public void emitToString(Target target, Constructor constructor) {
81          target.write("      @Override\n");
82          target.write("      public String toString() {\n");
83          target.write("         return \"" + constructor.name);
84          if (!constructor.args.isEmpty()) {
85              target.write("(");
86              boolean first = true;
87              for (Arg arg : constructor.args) {
88                  if (first) {
89                      first = false;
90                  } else {
91                      target.write(", ");
92                  }
93                  target.write(arg.name + " = \" + " + arg.name + " + \"");
94              }
95              target.write(")");
96          }
97          target.write("\";\n");
98          target.write("      }");
99      }

101     /* (non-Javadoc)
102      * @see pogofish.jadt.emitter.ClassBodyEmitter#emitEquals(pogofish.jadt.emitter.Target, pogofish.jadt.ast.Constructor)
103      */
104     @Override
105     public void emitEquals(final Target target, Constructor constructor) {
106         target.write("      @Override\n");
107         target.write("      public boolean equals(Object obj) {\n");
108         target.write("         if (this == obj) return true;\n");
109         target.write("         if (obj == null) return false;\n");
110         target.write("         if (getClass() != obj.getClass()) return false;\n");
111         if (!constructor.args.isEmpty()) {
112             target.write("         " + constructor.name + " other = (" + constructor.name + ")obj;\n");
113             
114             for (final Arg arg : constructor.args) {
115                 arg.type.accept(new Type.Visitor<Void>(){
116                     @Override
117                     public Void visit(Ref x) {
118                         x.type.accept(new RefType.Visitor<Void>() {
119                             @Override
120                             public Void visit(ClassType x) {
121                                 target.write("         if (" + arg.name + " == null) {\n");
122                                 target.write("            if (other." + arg.name + " != null) return false;\n");
123                                 target.write("         } else if (!" + arg.name + ".equals(other." + arg.name + ")) return false;\n");
124                                 return null;
125                             }
126 
127                             @Override
128                             public Void visit(ArrayType x) {
129                                 target.write("         if (!java.util.Arrays.equals(" + arg.name + ", other." + arg.name + ")) return false;\n");
130                                 return null;
131                             }});
132                         return null;
133                     }
134 
135                     @Override
136                     public Void visit(Primitive x) {
137                         target.write("         if (" + arg.name + " != other." + arg.name + ") return false;\n");
138                         return null;
139                     }});
140             }
141         }
142         target.write("         return true;\n");
143         target.write("      }");
144     }

146     /* (non-Javadoc)
147      * @see pogofish.jadt.emitter.ClassBodyEmitter#emitHashCode(pogofish.jadt.emitter.Target, pogofish.jadt.ast.Constructor)
148      */
149     @Override
150     public void emitHashCode(final Target target, Constructor constructor) {
151         target.write("      @Override\n");
152         target.write("      public int hashCode() {\n");
153         if (constructor.args.isEmpty()) {
154             target.write("          return 0;\n");
155         } else {
156             target.write("          final int prime = 31;\n");
157             target.write("          int result = 1;\n");
158             for (final Arg arg : constructor.args) {
159                 arg.type.accept(new Type.Visitor<Void>(){
160                     @Override
161                     public Void visit(Ref x) {
162                         x.type.accept(new RefType.Visitor<Void>() {
163                             @Override
164                             public Void visit(ClassType x) {
165                                 target.write("          result = prime * result + ((" + arg.name + " == null) ? 0 : " + arg.name + ".hashCode());\n");                
166                                 return null;
167                             }
168 
169                             @Override
170                             public Void visit(ArrayType x) {
171                                 target.write("          result = prime * result + java.util.Arrays.hashCode(" + arg.name + ");\n");                
172                                 return null;
173                             }});
174                         return null;
175                     }
176 
177                     @Override
178                     public Void visit(Primitive x) {
179                         x.type.accept(new PrimitiveType.VisitorWithDefault<Void>() {
180 
181                             @Override
182                             public Void visit(IntType x) {
183                                 target.write("          result = prime * result + " + arg.name + ";\n");
184                                 return null;
185                             }
186 
187                             @Override
188                             public Void getDefault(PrimitiveType x) {
189                                 target.write("          result = prime * result + (int)" + arg.name + ";\n");                    
190                                 return null;
191                             }});
192                         return null;
193                     }});
194             }
195             target.write("          return result;\n");
196         }
197         target.write("      }");
198     }

201     private String constructorArg(Arg arg, boolean withType) {
202         return withType ? (Printer.print(arg.type) + " " + arg.name) : arg.name;
203     }
