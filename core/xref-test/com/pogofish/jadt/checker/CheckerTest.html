39  public class CheckerTest {
40      /**
41       * Check with duplicate data types
42       */
43      @Test
44      public void testDuplicateDataType() {
45          final Checker checker = new StandardChecker();
46          final DataType dataType = new DataType("Foo", Util.<String>list(), list(new Constructor("Foo", Util.<Arg>list())));
47          final Doc doc = new Doc("CheckerTest", "", Util.<String>list(), list(dataType, dataType));
48          final List<SemanticError> errors = checker.check(doc);
49          assertEquals(1, errors.size());
50          assertTrue(errors.contains(_DuplicateDataType(dataType.name)));
51      }
52      
53      /**
54       * Check with duplicate constructors for one data type
55       */
56      @Test
57      public void testDuplicateConstructor() {
58          final Checker checker = new StandardChecker();
59          final Constructor constructor = new Constructor("Bar", Util.<Arg>list());
60          final DataType dataType = new DataType("Foo", Util.<String>list(), list(constructor, constructor));
61          final Doc doc = new Doc("CheckerTest", "", Util.<String>list(), list(dataType));
62          final List<SemanticError> errors = checker.check(doc);
63          assertEquals(1, errors.size());
64          assertTrue(errors.contains(_DuplicateConstructor(dataType.name, constructor.name)));
65      }
