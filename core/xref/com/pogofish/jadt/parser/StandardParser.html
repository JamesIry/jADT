61  public class StandardParser implements Parser {
62  	private static final Logger logger = Logger.getLogger(StandardParser.class.toString());

65      /* (non-Javadoc)
66       * @see sfdc.adt.IParser#parse(java.lang.String, java.io.Reader)
67       */
68      @Override
69      public ParseResult parse(Source source)  {
70      	logger.fine("Parsing " + source.getSrcInfo());
71      	final BufferedReader reader = source.createReader();
72      	try {
73              final Tokenizer tokenizer = new Tokenizer(source.getSrcInfo(), reader);
74              final Impl impl = new Impl(tokenizer);
75              final Doc doc = impl.doc();
76              return new ParseResult(doc, impl.errors);
77      	} finally {
78      	    new IOExceptionAction<Void>() {
79                  @Override
80                  public Void doAction() throws IOException {
81                      reader.close();
82                      return null;
83                  }
84              }.execute();
85      	}
86      }

95           * Tokenizer to be parsed

97          private final Tokenizer tokenizer;

99          /**
100          * Whether the parser is currently recovering from an error. If true, new errors
101          * will not be added to the errors list

103         private boolean recovering = false;

108         List<SyntaxError> errors = new ArrayList<SyntaxError>();

123         /**
124          * Creates a Parser.Impl that will parse the given tokenizer using the srcInfo for error reporting
125          * 
126          * @param srcInfo String information about the source that is used when throwing a syntax exception
127          * @param tokenizer Tokenizer to be parsed

129         public Impl(Tokenizer tokenizer) {
130             this.tokenizer = tokenizer;
131         }

133         /**
134          * Peeks at the next available token without removing it, returning true if it's in the given set
135          * 
136          * @return
137          */
138         private boolean peek(Set<TokenType> expected) {
139             final TokenType token = tokenizer.getTokenType();
140             tokenizer.pushBack();
141             return expected.contains(token);
142         }
143         
144         /**
145          * Peeks at the next available token without removing, returning true if it's in the set of punctuation tokens
146          * @return
147          */
148         private boolean peekPunctuation() {
149             return peek(tokenizer.punctuation);
150         }
151 
152         /**
153          * If the next token type is the expected token type then it is consumed and true is returned
154          * otherwise false is returned and the token is not consumed
155          * @param expected
156          */
157         private boolean accept(TokenType expected) {
158             final TokenType token = tokenizer.getTokenType();
159             if (token.equals(expected)) {
160                 recovering = false;
161                 return true;
162             } else {
163                 tokenizer.pushBack();
164                 return false;
165             }
166         }
167         
168         /**
169          * Consume any token type and return the symbol.  Used during error recovery
170          */
171         private String consumeAnything() {
172             tokenizer.getTokenType();
173             return tokenizer.lastSymbol();
174         }

179          * @return Doc

184             return new Doc(tokenizer.srcInfo(), pkg(), imports(), dataTypes());

187         /** 
188          * Gets an optional package declaration "pacakage" packageName
189          * 
190          * @return String the package name, or, because package is optional returns an empty string ""
191          */
192         public String pkg() {
193         	logger.finer("Parsing package");
194             if (accept(TokenType.PACKAGE)) {
195                 final String name = packageName();
196                 logger.finer("package " + name);
197                 return name;
198             } else {
199             	logger.finer("no package");
200                 return "";
201             }
202         }

204         /**
205          * Recognizes a required token that is a valid package name
206          * 
207          * @return String the package name
208          */
209         private String packageName() {
210             if (accept(TokenType.IDENTIFIER)) {
211                 return tokenizer.lastSymbol();
212             } else if (accept(TokenType.DOTTED_IDENTIFIER)) {
213                 return tokenizer.lastSymbol();
214             } else {
215                 error("a package name");
216                 if (peek(set(TokenType.IMPORT, TokenType.PACKAGE, TokenType.EOF))) {
217                     // if the token was import or package then assume they just forgot the package name
218                     // and move on
219                     return "NO_PACKAGE_NAME" + nextId();                    
220                 } else {
221                     // otherwise assume that they used a keyword as a package name.  Consume it and move on
222                     return "BAD_PACKAGE_NAME_" + consumeAnything() + nextId();
223                 }
224             }
225         }

227         /**
228          * Parses an optional list of imports which is ("import" packageName)*
229          * 
230          * @return List<String> possibly empty list of imports
231          */
232         public List<String> imports() {
233         	logger.finer("Parsing imports");
234             final List<String> imports = new ArrayList<String>();
235             
236             while (accept(TokenType.IMPORT)) {
237             	final String name = packageName();
238             	logger.finer("import " + name);
239                 imports.add(name);
240             }
241             return imports;
242

245         /**
246          * Parses a required list of datatypes which is dataType (dataType)*
247          * 
248          * @return List<DataType> non empty list of DataTypes
249          */
250         public List<DataType> dataTypes() {
251         	logger.finer("Parsing datatypes");
252             final List<DataType> dataTypes = new ArrayList<DataType>();
253 
254             dataTypes.add(dataType());
255             
256             while (!accept(TokenType.EOF)) {
257                 dataTypes.add(dataType());
258             }
259             return Collections.unmodifiableList(dataTypes);
260         }

262         /**
263          * Parses a dataType which is (name typeArguments? "=" constructors)
264          * 
265          * @return DataType
266          */
267         public DataType dataType() {
268         	logger.finer("Parsing datatype");
269         	final String name;
270         	if (accept(TokenType.IDENTIFIER)) {
271         	    name = tokenizer.lastSymbol();
272         	} else {
273                 error("a data type name");               
274         	    if (peekPunctuation()) {
275         	        // if the token was punctuation then they just forgot a name, make one up and move on
276                     name = "NO_DATA_TYPE_NAME" + nextId();        	        
277         	    } else {
278         	        // otherwise let's consume the token as if it was the name and try to move on
279         	        name = "BAD_DATA_TYPE_NAME_" + consumeAnything() + nextId();
280         	    }
281         	}
282 
283             logger.finer("datatype " + name );
284             final List<String> typeArguments = typeArguments();
285 
286             if (!accept(TokenType.EQUALS)) {
287                 // with no equals we'll pretend it was there and try to move on
288                 error("'='");
289             }
290 
291             return new DataType(name, typeArguments, constructors());
292         }

294         /**
295          * Parses an optional type arguments which is ('<' typeArgument (',' typeArgument)* '>')?
296          * 
297          * @return list of string names of type arguments. Empty if none
298          */
299         public List<String> typeArguments() {
300             logger.finest("parsing type arguments");
301             final List<String> typeArguments = new ArrayList<String>();
302         	
303             if(accept(TokenType.LANGLE)) {
304             	typeArguments.add(typeArgument());
305 	            while (accept(TokenType.COMMA)) {
306 	            	typeArguments.add(typeArgument());
307 	            }
308 	            boolean done = false;
309 	            while (!done) {	                
310     	            if (accept(TokenType.RANGLE)) {
311     	            	done = true;
312     	            } else {
313                         error("'>'");
314                         if (peekPunctuation()) {
315                             // just forget the >, move on, leaving the equals to be found later
316                             done = true;
317     
318                         } else {
319                             // probably just missing a comma, consume whatever you have, add it to the list,
320                             // and keep looking for the >
321                             typeArguments.add(typeArgument());
322                         }
323     	            }
324 	            }
325             } else {
326             	logger.finest("no type arguments");
327             }
328             return typeArguments;
329         }

331         /**
332          * A required type argument which is just any identifier
333          */
334         public String typeArgument() {
335             if (accept(TokenType.IDENTIFIER)) {
336                 final String name = tokenizer.lastSymbol();
337                 logger.finest("type argument " + name);
338                 return tokenizer.lastSymbol();  
339             } else {
340                 error("a type parameter");
341                 if (peekPunctuation()) {
342                     // missing a name, leave the punctuation for later
343                     return "NO_TYPE_ARGUMENT" + nextId();
344                 } else {
345                     // consume whatever you have and pretend it's a type argument
346                     return "BAD_TYPE_ARGUMENT_" + consumeAnything() + nextId();
347                 }                
348             }
349         }

357         	logger.finer("parsing constructors");
358             final List<Constructor> constructors = new ArrayList<Constructor>();
359             constructors.add(constructor());

366         /**
367          * Parses a required constructor which is constructorName args
368          * 
369          * @return Constructor
370          */
371         public Constructor constructor() {
372         	logger.finer("parsing constructor");
373         	final String name;
374         	
375             if (accept(TokenType.IDENTIFIER)) {
376                 name = tokenizer.lastSymbol();
377             } else {
378                 error("a constructor name");
379                 if (peekPunctuation()) {
380                     // we're proably missing a constructor name.  Leave the punctuatino somebody else
381                     name = "NO_CONSTRUCTOR_NAME" + nextId();
382                 } else {
383                     // otherwise consume whatever we have as the name
384                     name = "BAD_CONSTRUCTOR_NAME_" + consumeAnything() + nextId(); 
385                 }
386             }
387             
388             logger.finer("constructor " + name);
389             return new Constructor(name, args());
390         }

392         /** 
393          * Parses an optional list of constructor args which is ("(" arg ("," arg)* ")")?
394          * 
395          * @return List<Arg> non-empty list of args
396          */
397         public List<Arg> args() {
398             final List<Arg> args = new ArrayList<Arg>();
399             if (accept(TokenType.LPAREN)) {
400                 logger.finest("parsing args");
401                 args.add(arg());
402                 while (accept(TokenType.COMMA)) {
403                     args.add(arg());
404                 }
405                 boolean done = false;
406                 while (!done) {
407                     if (accept(TokenType.RPAREN)) {
408                         done = true;
409                     } else {
410                         error("')'");
411                         if (peekPunctuation()) {
412                             // probably just forgot a closing paren, we're done, leave the punctuation for somebody else
413                             done = true;
414                         } else {
415                             // probably just missing a comma, consume whatever you have, add it to the list,
416                             // and keep looking for the >
417                             args.add(arg());
418                         }
419 
420                     }
421                 }
422             }           
423             return Collections.unmodifiableList(args);
424         }

426         /**
427          * parses a single required constructor argument which is a list of modifiers followed by a type followed by a name
428          * 
429          * @return Arg
430          */
431         public Arg arg() {
432             final List<ArgModifier> modifiers = argModifiers();
433             
434             final Type type = type();            
435 
436             final String name;
437             if (accept(TokenType.IDENTIFIER)) {
438                 name = tokenizer.lastSymbol();
439                 logger.finest("arg " + name);
440             } else {             
441                 error("an argument name");
442                 if (peekPunctuation()) {
443                     // Probably just missing a name.  Leave the token for somebody else
444                     name = "NO_ARG_NAME" + nextId();
445                 } else {
446                     // it's not a good name but is used as one, consume it
447                     name = "BAD_ARG_NAME_" + consumeAnything() + nextId();
448                 }
449             }
450             return new Arg(modifiers, type, name);
451         }

453         /**
454          * Parses a possibly empty list of arg modifiers
455          */
456         public List<ArgModifier> argModifiers() {
457             final List<ArgModifier> modifiers = Util.<ArgModifier>list();
458 
459             ArgModifier modifier;
460             do {
461                 modifier = argModifier();
462                 if (modifier != null) {
463                     modifiers.add(modifier);
464                 }
465             } while (modifier != null);
466 
467             return modifiers;
468         }

480         /** 
481          * Returns a required class or array type, giving a syntax error on a primitive
482          * 
483          * @return a RefType
484          */
485         public RefType refType() {
486             return type().match(new Type.MatchBlock<RefType>() {
487 
488                 @Override
489                 public RefType _case(Ref x) {
490                     return x.type;
491                 }
492 
493                 @Override
494                 public RefType _case(Primitive x) {
495                     final String primitiveName = ASTPrinter.print(x);
496                     error ("an array or class type", primitiveName);
497                     // we got a primitive when we expected an array or class, so make up a class
498                     return _ClassType("BAD_CLASS_" + primitiveName + nextId(), Util.<RefType>list());
499 
500                 }});
501         }

503         /**
504          * Returns a required type where a type is a primitive or classType wrapped in an array
505          * 
506          * @return Type
507          */
508         public Type type() {
509             final PrimitiveType primitive = primitiveType();
510             final Type type = primitive == null ? _Ref(classType()) : _Primitive(primitive);
511            	logger.finest("type " + type);
512             return array(type);
513         }

515         /**
516          * Recursively wraps a type in ArrayType based on the number of "[]" pairs immediately in the stream
517          * @param heldType last type parsed
518          * @return either the original held type or that type wrapped in ArrayType
519          */
520         public Type array(Type heldType) {
521             if (accept(TokenType.LBRACKET)) {
522                 if (!accept(TokenType.RBRACKET)) {
523                     error("']'");
524                     // leave whatever is lying around for somebody else to figure out                    
525                 }
526                 return array(_Ref(_ArrayType(heldType)));
527             }
528             return heldType;            
529         }

531         /**
532          * Returns a required classType, which is className ("<" refType ("," refType)* ">")?
533          * 
534          * @return RefType
535          */
536         public RefType classType() {
537             final String className = className();
538             
539             final List<RefType> typeArguments = Util.<RefType>list();
540             if (accept(TokenType.LANGLE)) {
541                 typeArguments.add(refType());
542                 while(accept(TokenType.COMMA)) {
543                     typeArguments.add(refType());
544                 }
545                 boolean done = false;
546                 while (!done) {
547                     if (accept(TokenType.RANGLE)) {
548                         done = true;
549                     } else {
550                         error("'>'");
551                         if (peekPunctuation()) {
552                             // got punctuation...um...pretend we're done and leave the punctuation for later
553                             done = true;
554                         } else {
555                             // got something else, try to treat it as a refType
556                             typeArguments.add(refType());
557                         }
558                     }
559                 }
560             }
561             return _ClassType(className, typeArguments);
562         }

564         /**
565          * Returns a class name which is either an identifier or a dotted identifier
566          */
567         public String className() {
568             if (accept(TokenType.IDENTIFIER) || (accept(TokenType.DOTTED_IDENTIFIER))) {
569                 return tokenizer.lastSymbol();
570             } else {
571                 error("a class name");
572                 if (peekPunctuation()) {
573                     // most punctaion probably just means a missing name.  Leave the punctuation for somebody else
574                     return "NO_CLASS_NAME" + nextId();
575                 } else {
576                     // otherwise consume whatever is there as if it were the name
577                     return "BAD_CLASS_NAME_" + consumeAnything() + nextId();
578                 }
579             }           
580         }

584          *

587         public PrimitiveType primitiveType() {
588             if (accept(TokenType.BOOLEAN)) {
589                 return(_BooleanType()); 
590             } else if (accept(TokenType.CHAR)) {
591                 return(_CharType()); 
592             } else if (accept(TokenType.SHORT)) {
593                 return(_ShortType()); 
594             } else if (accept(TokenType.INT)) {
595                 return(_IntType()); 
596             } else if (accept(TokenType.LONG)) {
597                 return(_LongType()); 
598             } else if (accept(TokenType.FLOAT)) {
599                 return(_FloatType()); 
600             } else if (accept(TokenType.DOUBLE)) {
601                 return(_DoubleType());
602             } else {
603                 return null;
604             }            
605         }

607         /**
608          * If not currently recovering, adds an error to the errors list and sets recovering to true.
609          * Actual is assumed to be the last symbol from the tokenizer.
610 
611          * @param expected the kind of thing expected
612          */
613         private void error(String expected) {
614             error(expected, tokenizer.lastSymbol());
615         }
616         
617         /**
618          * If not currently recovering, adds an error to the errors list and sets recovering to true
619          * @param expected the kind of thing expected
620          */
621         private void error(String expected, String actual) {
622             if (!recovering) {
623                 final String outputString = "<EOF>".equals(actual) ? actual : "'" + actual + "'";
624                 errors.add(SyntaxError._UnexpectedToken(expected, outputString, tokenizer.lineno()));
625                 recovering = true;
626             }
627         }
