9   package com.pogofish.jadt.sink;
10  
11  import java.io.File;
12  import java.io.FileOutputStream;
13  import java.io.IOException;
14  import java.io.OutputStreamWriter;
15  import java.io.Writer;
16  
17  import com.pogofish.jadt.util.ExceptionAction;
18  import com.pogofish.jadt.util.Util;
19  
20  
21  /**
22   * Sink that writes to a specified file in UTF-8
23   *
24   * @author jiry
25   */
26  public class FileSink implements Sink {
27      private final Writer writer;
28      final File outputFile;
29  
30      @Override
31  	public String getInfo() {
32  		return outputFile.getAbsolutePath();
33  	}
34  
35  	/**
36       * Creates a FileSink based on a complete file name
37       * 
38       * @param outputFileName String full name of the file to be output
39       */
40      public FileSink(final String outputFileName) {
41          super();
42          outputFile = new File(outputFileName);
43  
44          writer = Util.execute(new ExceptionAction<Writer>() {
45              @Override
46              public Writer doAction() throws IOException {
47                  final File parentDir = outputFile.getParentFile();
48                  parentDir.mkdirs();
49  
50                  outputFile.createNewFile();
51  
52                  return new OutputStreamWriter(new FileOutputStream(outputFile), "UTF-8");
53              }
54          });
55      }
56  
57      @Override
58      public void write(final String data) {
59          Util.execute(new ExceptionAction<Writer>() {
60  
61              @Override
62              public Writer doAction() throws IOException {
63                  writer.write(data);
64                  return null;
65              }});
66      }
67  
68      @Override
69      public void close() {
70          Util.execute(new ExceptionAction<Writer>() {
71  
72              @Override
73              public Writer doAction() throws IOException {
74                  writer.close();
75                  return null;
76              }});
77      }
78  }
