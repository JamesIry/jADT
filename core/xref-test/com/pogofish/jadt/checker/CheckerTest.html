49      /**
50       * Check with duplicate data types
51       */
52      @Test
53      public void testDuplicateDataType() {
54          final Checker checker = new StandardChecker();
55          final DataType dataType = new DataType(NO_COMMENTS, "Foo", Util.<String>list(), list(new Constructor(NO_COMMENTS, "Foo", Util.<Arg>list())));
56          final Doc doc = new Doc("CheckerTest", Pkg._Pkg(NO_COMMENTS, ""), NO_IMPORTS, list(dataType, dataType));
57          final List<SemanticError> errors = checker.check(doc);
58          assertEquals(1, errors.size());
59          assertTrue(errors.contains(_DuplicateDataType(dataType.name)));
60      }
61      
62      /**
63       * Check with duplicate constructors for one data type
64       */
65      @Test
66      public void testDuplicateConstructor() {
67          final Checker checker = new StandardChecker();
68          final Constructor constructor = new Constructor(NO_COMMENTS, "Bar", Util.<Arg>list());
69          final DataType dataType = new DataType(NO_COMMENTS, "Foo", Util.<String>list(), list(constructor, constructor));
70          final Doc doc = new Doc("CheckerTest", Pkg._Pkg(NO_COMMENTS, ""), NO_IMPORTS, list(dataType));
