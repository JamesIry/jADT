113     /**
114      * Test a reasonably fully document with pacakge, imports, and datatypes
115      */
116     @Test
117     public void testFull() {
118         final Doc doc = new Doc("EmitterTest", "some.package", list("wow.man", "flim.flam"), list(
119                 new DataType("FooBar", Util.<String>list(), list(
120                         new Constructor("Foo", list(
121                                 new Arg(_Primitive(_IntType()), "yeah"),
122                                 new Arg(_Ref(_ClassType("String", Util.<RefType>list())), "hmmm")
123                         )),
124                         new Constructor("Bar", Util.<Arg>list())
125                 )),
126                 new DataType("Whatever", Util.<String>list(), list(
127                         new Constructor("Whatever", Util.<Arg>list())
128                 ))
129                 
130         ));
131         final StringTargetFactory factory = new StringTargetFactory("whatever");
132         final DocEmitter emitter = new StandardDocEmitter(new DummyDataTypeEmitter());
133         emitter.emit(factory, doc);
134         final Map<String, String> results = factory.getResults();
135         assertEquals("Got the wrong number of results", 2, results.size());
136         final String foobar = results.get("some.package.FooBar");
137         assertEquals(FULL_HEADER+FOOBAR, foobar);
138         assertEquals(FULL_HEADER+WHATEVER, results.get("some.package.Whatever"));
139     }
