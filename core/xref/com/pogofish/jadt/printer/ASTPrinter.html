33  /**
34   * Pretty printer for the jADT AST.  Note that in the context of pretty printers "print" just means "make a nice looking string"
35   *
36   * @author jiry
37   */
38  public class ASTPrinter  {
39      /**
40       * ( "package" packageName )?
41       * (
42       * ( "import" import \n)*
43       * )?
44       * dataType
45       * (dataType\n)*
46       * 
47       * @param doc Document to be printed
48       * @return pretty string
49       */
50      public static String print(Doc doc) {
51          final StringBuilder builder = new StringBuilder(doc.pkg.isEmpty() ? "" : ("package " + doc.pkg + "\n\n"));
52          if (!doc.imports.isEmpty()) {
53              for (String imp : doc.imports) {
54                  builder.append("import " + imp + "\n");
55              }
56              builder.append("\n");
57          }
58          for (DataType dataType : doc.dataTypes) {
59              builder.append(print(dataType));
60              builder.append("\n");
61          }
62          return builder.toString();  
63          
64      }

66      /**
67       * Prints a dataType as
68       * dataType = constructor
69       * (\n  | constructor)*
70       * 
71       * @param dataType DataType to be printed
72       * @return pretty string
73       */
74      public static String print(DataType dataType) {
75          final StringBuilder builder = new StringBuilder(dataType.name + " =\n    ");
76          boolean first = true;
77          for (Constructor constructor : dataType.constructors) {
78              if (first) {
79                  first = false;
80              } else {
81                  builder.append("\n  | ");
82              }
83              builder.append(print(constructor));
84          }
85          return builder.toString();
86      }

88      /**
89       * Prints a constructor as
90       * name("("arg(", "arg)*")")?
91       * 
92       * @param constructor
93       * @return pretty string
94       */
95      public static String print(Constructor constructor) {
96          final StringBuilder builder = new StringBuilder(constructor.name);
97          if (! constructor.args.isEmpty()) {
98              builder.append("(");
99              boolean first = true;
100             for (Arg arg : constructor.args) {
101                 if (first) {
102                     first = false;
103                 } else {
104                     builder.append(", ");
105                 }
106                 builder.append(print(arg));
107             }
108             builder.append(")");
109         }
110         return builder.toString();
111     }

113     /**
114      * prints an arg as
115      * type name
116      * 
117      * @param arg Arg to be printed
118      * @return pretty string
119      */
120     public static String print(Arg arg) {
121         return print(arg.modifiers) + print(arg.type) + " " + arg.name;
122     }

124     /**
125      * Prints a list of arg modifiers
126      * @param modifiers list of arg modifiers
127      * @return string consisting of those modifiers, with a trailing space
128      */
129     public static String print(List<ArgModifier> modifiers) {
130         final StringBuilder builder = new StringBuilder();
131         if (modifiers.contains(ArgModifier._Final())) {
132             builder.append(print(ArgModifier._Final()));
133             builder.append(" ");
134         }
135         return builder.toString();
136     }

138     /**
139      * Prints a type as either a ref type or a primitive
140      * 
141      * @param type Type to be printed
142      * @return pretty string
143      */
144     public static String print(Type type) {
145         return type.match(new Type.MatchBlock<String>(){
146             @Override
147             public String _case(Ref x) {
148                 return print(x.type);
149             }
150 
151             @Override
152             public String _case(Primitive x) {
153                 return print(x.type);
154             }
155             
156         });
157     }

159     /**
160      * Prints a type as either an array or class type. 
161      * 
162      * For a an array, it prints as
163      * type[]
164      * 
165      * For a class it prints as
166      * className("<" refType(", " refType)*">")?
167      * 
168      * @param type RefType to be printed
169      * @return pretty string
170      */
171     public static String print(RefType type) {
172         return type.match(new RefType.MatchBlock<String>() {
173 
174             @Override
175             public String _case(ClassType x) {
176                 final StringBuilder builder = new StringBuilder(x.baseName);
177                 if (!x.typeArguments.isEmpty()) {
178                     builder.append("<");
179                     boolean first = true;
180                     for (RefType typeArgument : x.typeArguments) {
181                         if (first) {
182                             first = false;
183                         } else {
184                             builder.append(", ");
185                         }
186                         builder.append(print(typeArgument));
187                     }
188                     builder.append(">");
189                 }
190                 return builder.toString();
191             }
192 
193             @Override
194             public String _case(ArrayType x) {
195                 return print(x.heldType) + "[]";
196             }});
197     }
