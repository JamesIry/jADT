60      /**
61       * Convenient factory method to create a complete standard configuration
62       * 
63       * @return Driver configured with all the Standard bits
64       */
65      public static JADT standardConfigDriver() {
66      	logger.fine("Using standard configuration.");
67          final SourceFactory sourceFactory = new FileSourceFactory();
68          final ClassBodyEmitter classBodyEmitter = new StandardClassBodyEmitter();
69          final ConstructorEmitter constructorEmitter = new StandardConstructorEmitter(classBodyEmitter);
70          final DataTypeEmitter dataTypeEmitter = new StandardDataTypeEmitter(classBodyEmitter, constructorEmitter);
71          final DocEmitter docEmitter = new StandardDocEmitter(dataTypeEmitter);      
72          final Parser parser = new StandardParser();
73          final Checker checker = new StandardChecker();
74          final TargetFactoryFactory factoryFactory = new FileTargetFactoryFactory();
75          
76          return new JADT(sourceFactory, parser, checker, docEmitter, factoryFactory);
77      }

79      /**
80       * Constructs a driver with the given components.
81       * 
82       * @param parser Parser to read jADT files
83       * @param checker Checker to validate jADT structures
84       * @param emitter Emitter to spit out java files
85       */
86      public JADT(SourceFactory sourceFactory, Parser parser, Checker checker, DocEmitter emitter, TargetFactoryFactory factoryFactory) {
87          super();
88          this.sourceFactory = sourceFactory;
89          this.parser = parser;
90          this.emitter = emitter;
91          this.checker = checker;
92          this.factoryFactory = factoryFactory;
93      }

95      /**
96       * Do the jADT thing based on an array of String args.  There must be 2 and the must be the source file and destination directory
97       * 
98       * @param args
99       */
100     public void parseAndEmit(String[] args) {
101     	logger.finest("Checking command line arguments.");
102         if (args.length != 2) {
103         	final String version = new Version().getVersion();
104         	logger.info("jADT version " + version + ".");
105         	logger.info("Not enough arguments provided to jADT");
106         	logger.info("usage: java sfdc.adt.JADT [source file] [output directory]");
107             throw new IllegalArgumentException("\njADT version " + version + "\nusage: java sfdc.adt.JADT [source file] [output directory]");
108         }
109         
110         final String srcFileName = args[0];
111         final String destDirName = args[1];
112 
113         parseAndEmit(srcFileName, destDirName);        
114     }
115 
116     /**
117      * Do the jADT thing given the srceFileName and destination directory
118      * 
119      * @param srcFileName full name of the source directory
120      * @param destDir full name of the desintation directory (trailing slash is optional)
121      */
122     public void parseAndEmit(String srcFileName, String destDir) {    	
123     	final String version = new Version().getVersion();
124     	logger.info("jADT version " + version + ".");
125     	logger.info("Will read from srcFile " + srcFileName);
126     	logger.info("Will write to destDir " + destDir);
127    	
128         final Source source = sourceFactory.createSource(srcFileName);
129         try {
130             final Doc doc = parser.parse(source);
131             final Set<SemanticException> errors = checker.check(doc);
132             if (!errors.isEmpty()) {
133                 throw new SemanticExceptions(errors);
134             }
135             final TargetFactory targetFactory = factoryFactory.createTargetFactory(destDir);
136             emitter.emit(targetFactory, doc);
137         } finally {
138             source.close();
139         }
140     }
