27  /**
28   * Modeled after the interface of a subset of Java's StreamTokenizer, but returns TokenTypes designed for

34      /**
35       * Regex for one piece of an identifier
36       */
37      private static final String IDENTIFIER_CHUNK = "\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*";
38      /**
39       * Regex for a valid java identifier
40       */
41      private static final Pattern IDENTIFIER_REGEX = Pattern.compile(IDENTIFIER_CHUNK);
42      /**
43       * Regex for a valid dotted java identifier, e.g. in a package name
44       */
45      private static final Pattern DOTTED_IDENTIFIER_REGEX = Pattern.compile("(" + IDENTIFIER_CHUNK + "\\.)+" + IDENTIFIER_CHUNK);

110     private final String srcInfo;

112     /**
113      * Current implementation uses a StreamTokenizer under the hood
114      */
115     private final StreamTokenizer tokenizer;
116 
117     /**
118      * The last symbol recognized
119      */
120     private String symbol = null;

122     /**
123      * Constructs a Tokenizer that will tokenize the specified Reade
124      * 
125      * @param reader the reader with the JADT source to be tokenized
126      */
127     public Tokenizer(Source source) {
128         srcInfo = source.getSrcInfo();
129         tokenizer = new StreamTokenizer(source.getReader());
130         tokenizer.resetSyntax();
131         tokenizer.wordChars(0, Integer.MAX_VALUE);
132         tokenizer.ordinaryChar('<');
133         tokenizer.ordinaryChar('>');
134         tokenizer.ordinaryChar('=');
135         tokenizer.ordinaryChar('(');
136         tokenizer.ordinaryChar(')');
137         tokenizer.ordinaryChar(',');
138         tokenizer.ordinaryChar('|');
139         tokenizer.ordinaryChar('[');
140         tokenizer.ordinaryChar(']');
141         tokenizer.whitespaceChars(' ', ' ');
142         tokenizer.whitespaceChars('\t', '\t');
143         tokenizer.whitespaceChars('\n', '\n');
144         tokenizer.whitespaceChars('\r', '\r');
145         tokenizer.eolIsSignificant(false);
146         tokenizer.ordinaryChar('/');
147         tokenizer.ordinaryChar('*');
148         tokenizer.slashSlashComments(true);
149         tokenizer.slashStarComments(true);
150     }

152     /**
153      * Get the next Token
154      * 
155      * @return a Token
156      */
157     private TokenType getTokenType() {
158         final int tokenType;
159         try {
160             tokenType = tokenizer.nextToken();
161         } catch (IOException e) {
162             throw new RuntimeException(e);
163         }
164         
165         switch (tokenType) {
166         case StreamTokenizer.TT_EOF:
167             symbol = "<EOF>";
168             return TokenType.EOF;
169         case StreamTokenizer.TT_WORD:
170             symbol = tokenizer.sval;
171             if (KEYWORDS.containsKey(symbol)) {
172                 return KEYWORDS.get(symbol);
173             } else {
174                 final Matcher identifierMatcher = IDENTIFIER_REGEX.matcher(symbol);
175                 if (identifierMatcher.matches()) {
176                     return TokenType.IDENTIFIER;
177                 }
178                 final Matcher dottedIdentifierMatcher = DOTTED_IDENTIFIER_REGEX.matcher(symbol);
179                 if (dottedIdentifierMatcher.matches()) {
180                     return TokenType.DOTTED_IDENTIFIER;
181                 }
182                 return TokenType.UNKNOWN;
183             }
184         case '(':
185             symbol = "(";
186             return TokenType.LPAREN;
187         case ')':
188             symbol = ")";
189             return TokenType.RPAREN;
190         case '<':
191             symbol = "<";
192             return TokenType.LANGLE;
193         case '>':
194             symbol = ">";
195             return TokenType.RANGLE;
196         case '[':
197             symbol = "[";
198             return TokenType.LBRACKET;
199         case ']':
200             symbol = "]";
201             return TokenType.RBRACKET;
202         case '=':
203             symbol = "=";
204             return TokenType.EQUALS;
205         case ',':
206             symbol = ",";
207             return TokenType.COMMA;
208         case '|':
209             symbol = "|";
210             return TokenType.BAR;
211         default:
212             symbol = "" + (char)tokenType;
213             return TokenType.UNKNOWN;
214         }
215     }

217     /**
218      * Peeks at the next available token without removing it
219      * 
220      * @return
221      */
222     public TokenType peek() {
223         final TokenType tokenType = getTokenType();
224         tokenizer.pushBack();
225         return tokenType;
226     }
227 
228     /**
229      * If the next token type i
230      * @param expected
231      * @return
232      */
233     public boolean accept(TokenType expected) {
234         final TokenType token = getTokenType();
235         if (token.equals(expected)) {
236             return true;
237         } else {
238             tokenizer.pushBack();
239             return false;
240         }
241     }
