34  /**
35   * Interface for internal parser implementations used by the StandardParser
36   * The idea is that StandardParser provides a stateless facaade to the world
37   * while ParseImpl may be stateful
38   * 
39   * @author jiry
40   */
41  public interface ParserImpl {
42  
43      /**
44       * Get the srcInfo for the source that created this parser
45       */
46      public abstract String getSrcInfo();
47      
48      /**
49       * A jADT document has an optional package followed by imports and datatypes and ends with an end of file.
50       */
51      public abstract Doc doc() throws Exception;
52  
53      /**
54       * The package declaration (if it exists) is the keyword "package" followed by a package name.  No semicolon required.
55       */
56      public abstract Pkg pkg() throws Exception;
57  
58      /**
59       *     A jADT document may specify any number (including 0) of package names to import.
60       */
61      public abstract List<Imprt> imports() throws Exception;
62  
63      /**
64       * Each import is the keyword "import" followed by a package name.
65       */
66      public abstract Imprt singleImport() throws Exception;
67  
68      /**
69       * A package name used in a package declaration must be a valid Java package name.
70       */
71      public abstract String packageName() throws Exception;
72  
73      /**
74       * A package spec used in an import declaration.
75       */
76      public abstract String packageSpec() throws Exception;
77  
78      /**
79       *     A jADT document must have at least one datatype, but may have as many as you'd like.
80       */
81      public abstract List<DataType> dataTypes() throws Exception;
82  
83      /**
84       * Each datatype consists of a name, optional type arguments, "=" and a list of constructors.
85       */
86      public abstract DataType dataType() throws Exception;
87  
88      /**
89       * A data type name is any valid Java identifier not qualified with a package.
90       */
91      public abstract CommentedIdentifier dataTypeName() throws Exception;
92  
93      /**
94       * A data type's type arguments, if it has them, start with '<', have 1 or more arguments, and end with '>'.
95       */
96      public abstract List<String> typeArguments() throws Exception;
97  
98      /**
99       * A single type argument is any valid Java identifier, not qualified with a package.
100      */
101     public abstract String typeArgument() throws Exception;
102 
103     /**
104      * A data type must have 1 or more case constructors separated by '|'.
105      */
106     public abstract List<Constructor> constructors(List<JavaComment> comments) throws Exception;
107 
108     /**
109      * A case constructor is a name optionally followed by arguments.  If it has no arguments then it must not be followed by '()'.
110      */
111     public abstract Constructor constructor(List<JavaComment> comments) throws Exception;
112 
113     /**
114      * A case constructor name is any valid Java identifier not qualified by a package.
115      */
116     public abstract CommentedIdentifier constructorName() throws Exception;
117 
118     /**
119      * If a case constructor arguments then they must start with '(', have 1 ore more arguments separated by ',' and end with ')'.
120      */
121     public abstract List<Arg> args() throws Exception;
122 
123     /**
124      * A case constructor argument is a list of modifiers, a type, and then a name.
125      */
126     public abstract Arg arg() throws Exception;
127 
128     /**
129      * A case constructor argument may have 0 or more modifiers separated by spaces.
130      */
131     public abstract List<ArgModifier> argModifiers() throws Exception;
132 
133     /**
134      * Currently the only argument modifier allowed is 'final' but others will follow.
135      */
136     public abstract ArgModifier argModifier() throws Exception;
137 
138     /**
139      * An argument name is any valid Java identifier not qualified by a package
140      */
141     public abstract String argName() throws Exception;
142 
143     /**
144      * A type is either a class type or a primitive type wrapped in 0 or more levels of array
145      */
146     public abstract Type type() throws Exception;
147 
148     /**
149      * A refType is either a primitive type wrapped in one level or array or a class type and in either case
150      * may be wrapped in 0 or more additional levels of array
151      */
152     public abstract RefType refType() throws Exception;
153 
154     /**
155      * The pair '[]' modifies a previously mentioned type X to make it type X[]
156      */
157     public abstract void arrayTypeBrackets() throws Exception;
158 
159     /**
160      * A class type is a class name followed by actual type arguments
161      */
162     public abstract RefType classType() throws Exception;
163 
164     /**
165      * A class name is a valid Java class name that may be qualified by a package.
166      */
167     public abstract String className() throws Exception;
168 
169     /**
170      * If a type has type arguments then it will be '<' followed by 1 or more refTypes followed by '>'.
171      */
172     public abstract List<RefType> actualTypeArguments() throws Exception;
173 
174     /**
175      * A primitive type is any of the standard Java primitive types
176      */
177     public abstract PrimitiveType primitiveType() throws Exception;
178 
179     /**
180      * A dotted identifier is a series of 1 or more identifiers separated by '.'.
181      */
182     public abstract String dottedIdentifier(String expected)
183             throws Exception;
184 
185     /**
186      * A commented identifier may be any valid Java identifier
187      * and may be precedeed by java comments
188      */
189     public abstract CommentedIdentifier commentedIdentifier(String expected)
190             throws Exception;
191 
192     /**
193      * An identifier may be any valid Java identifier
194      */
195     public abstract String identifier(String expected) throws Exception;
196 
197     /**
198      * 'import'
199      * may be precedeed by java comments
200      */
201     public abstract List<JavaComment> importKeyword() throws Exception;
202 
203     /**
204      * 'package'
205      * may be precedeed by java comments
206      */
207     public abstract List<JavaComment> packageKeyword() throws Exception;
208 
209     /**
210      * 'final'
211      */
212     public abstract ArgModifier finalKeyword() throws Exception;
213 
214     /**
215      * 'boolean'
216      */
217     public abstract PrimitiveType booleanType() throws Exception;
218 
219     /**
220      * 'byte'
221      */
222     public abstract PrimitiveType byteType() throws Exception;
223 
224     /**
225      * 'char'
226      */
227     public abstract PrimitiveType charType() throws Exception;
228 
229     /**
230      * 'short'
231      */
232     public abstract PrimitiveType shortType() throws Exception;
233 
234     /**
235      * 'int'
236      */
237     public abstract PrimitiveType intType() throws Exception;
238 
239     /**
240      * 'long'
241      */
242     public abstract PrimitiveType longType() throws Exception;
243 
244     /**
245      * 'float'
246      */
247     public abstract PrimitiveType floatType() throws Exception;
248 
249     /**
250      * 'double'
251      */
252     public abstract PrimitiveType doubleType() throws Exception;
253 
254     /**
255      * '.'
256      */
257     public abstract void dot() throws Exception;
258 
259     /**
260      * ','
261      */
262     public abstract void comma() throws Exception;
263 
264     /**
265      * '('
266      */
267     public abstract void lparen() throws Exception;
268 
269     /**
270      * ')'
271      */
272     public abstract void rparen() throws Exception;
273 
274     /**
275      * '['
276      */
277     public abstract void lbracket() throws Exception;
278 
279     /**
280      * ']'
281      */
282     public abstract void rbracket() throws Exception;
283 
284     /**
285      * '<'
286      */
287     public abstract void langle() throws Exception;
288 
289     /**
290      * '>'
291      */
292     public abstract void rangle() throws Exception;
293 
294     /**
295      * '='
296      * may be precedeed by Java comments
297      */
298     public abstract List<JavaComment> equals() throws Exception;
299 
300     /**
301      * '|'
302      * may be precedeed by Java comments
303      */
304     public abstract List<JavaComment> bar() throws Exception;
305 
306     /**
307      * end of file
308      */
309     public abstract void eof() throws Exception;
310 
311     /**
312      * Errors that occured during parsing
313      */
314     public abstract List<SyntaxError> errors();
