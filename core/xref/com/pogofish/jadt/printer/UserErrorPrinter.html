29  /**
30   * Pretty printer for the jADT user errors.  Note that in the context of pretty printers "print" just means "make a nice looking string"
31   *
32   * @author jiry
33   */
34  public class UserErrorPrinter {
35      public static String print(UserError error) {
36          return error.match(new UserError.MatchBlock<String>() {
37  
38              @Override
39              public String _case(Semantic x) {
40                  return print(x.error);
41              }
42  
43              @Override
44              public String _case(Syntactic x) {
45                  return print(x.error);
46              }
47          });
48      }
49      
50      public static String print(SyntaxError error) {
51          return  "Found token '" + error.found + "' but expected '" + error.expected + "' at line " + error.line + ".";
52      }
53      
54      public static String print(SemanticError error) {
55          return error.match(new SemanticError.MatchBlock<String>() {
56              @Override
57              public String _case(DuplicateDataType x) {
58                  return "Cannot have two datatypes named " + x.dataTypeName + " in one jADT document.";
59              }
60  
61              @Override
62              public String _case(ConstructorDataTypeConflict x) {
63                  return "Data type " + x.dataTypeName + " cannot have a constructor with the name " + x.dataTypeName +".  Only single constructor data types may have constructors with the same name.";
64             }
65  
66              @Override
67              public String _case(DuplicateConstructor x) {
68                  return "Data type " + x.dataTypeName + " cannot have multiple constructors named " + x.constructorName + ".";
69             }
70  
71              @Override
72              public String _case(DuplicateArgName x) {
73                  return "Duplicated arg name " + x.argName + " in constructor " + x.constructorName + " of data type " + x.dataTypeName + ".";
74              }
75  
76              @Override
77              public String _case(DuplicateModifier x) {
78                  return "Duplicated modifier " + x.modifier +" on arg name " + x.argName + " in constructor " + x.constructorName + " of data type " + x.dataTypeName + ".";
79              }
80          });
81      }
82  }
