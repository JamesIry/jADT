16  package com.pogofish.jadt.checker;

18  import java.util.ArrayList;
19  import java.util.HashSet;
20  import java.util.List;
21  import java.util.Set;
22  import java.util.logging.Logger;
23  
24  import com.pogofish.jadt.ast.*;
25  import static com.pogofish.jadt.ast.SemanticError.*;
26  import com.pogofish.jadt.printer.ASTPrinter;

29  /**
30   * Standard checker that finds problems in Doc documents
31   *
32   * @author jiry
33   */
34  public class StandardChecker implements Checker {
35  	private static final Logger logger = Logger.getLogger(StandardChecker.class.toString());
36      
37      /**
38       * Checks a documents for duplicate dataType names and calls check(DataType) on each one
39       * 
40       * @param doc Doc to check
41       * @return Set<SemanticError> with the problems or empty set if there are none
42       */
43      @Override 
44      public List<SemanticError> check(Doc doc) {
45      	logger.fine("Checking semantic constraints in document from " + doc.srcInfo);
46          final List<SemanticError> errors = new ArrayList<SemanticError>();
47          final Set<String> dataTypeNames = new HashSet<String>();
48          for (DataType dataType : doc.dataTypes) {
49              if (dataTypeNames.contains(dataType.name)) {
50              	logger.info("Duplicate data type name " + dataType.name + ".");
51                  errors.add(_DuplicateDataType(dataType.name));
52              } else {
53                  dataTypeNames.add(dataType.name);
54              }
55              errors.addAll(check(dataType));
56          }
57          return errors;
58      }
59  
60      /**
61       * Checks a data type for duplicate constructor names or constructors having the same name
62       * as the data type
63       * 
64       * @param dataType DataType to check
65       * @return Set<SemanticError> with the problems or empty set if there are none
66       */
67      private List<SemanticError> check(DataType dataType) {
68      	logger.finer("Checking semantic constraints on datatype " + dataType.name);
69          final List<SemanticError> errors = new ArrayList<SemanticError>();
70          final Set<String> constructorNames = new HashSet<String>();
71          for(Constructor constructor : dataType.constructors) {
72          	logger.finest("Checking semantic constraints on constructor " + constructor.name + " in datatype " + dataType.name);
73              if (dataType.constructors.size() > 1 && dataType.name.equals(constructor.name)) {
74              	logger.info("Constructor with same name as its data type " + dataType.name + ".");
75                  errors.add(_ConstructorDataTypeConflict(dataType.name));
76              }
77              if (constructorNames.contains(constructor.name)) {
78              	logger.info("Two constructors with same name " + constructor.name + " in data type " + dataType.name + ".");
79                  errors.add(_DuplicateConstructor(dataType.name, constructor.name));
80              } else {
81                  constructorNames.add(constructor.name);
82              }
83              errors.addAll(check(dataType, constructor));
84          }
85          return errors;
86      }
87      
88      /**
89       * Check a constructor to make sure it does not duplicate any args and that no arg duplicates any modifiers
90       * 
91       */
92      private List<SemanticError> check(DataType dataType, Constructor constructor) {
93          logger.finer("Checking semantic constraints on data type " + dataType.name + ", constructor " + constructor.name);
94          final List<SemanticError> errors = new ArrayList<SemanticError>();
95          final Set<String> argNames = new HashSet<String>();
96          for (Arg arg : constructor.args) {
97              if (argNames.contains(arg.name)) {
98                  errors.add(_DuplicateArgName(dataType.name, constructor.name, arg.name));
99              } else {
100                 argNames.add(arg.name);
101             }
102             errors.addAll(check(dataType, constructor, arg));
103         }
104         
105         return errors;
106     }
107 
108     /**
109      * Checkt to make sure an arg doesn't have duplicate modifiers
110      */
111     private List<SemanticError> check(DataType dataType, Constructor constructor, Arg arg) {
112         logger.finest("Checking semantic constraints on data type " + dataType.name + ", constructor " + constructor.name);
113         final List<SemanticError> errors = new ArrayList<SemanticError>();
114         final Set<ArgModifier> modifiers = new HashSet<ArgModifier>();
115         for (ArgModifier modifier : arg.modifiers) {
116             if (modifiers.contains(modifier)) {
117                 final String modName = ASTPrinter.print(modifier);
118                 errors.add(_DuplicateModifier(dataType.name, constructor.name, arg.name, modName));
119             } else {
120                 modifiers.add(modifier);
121             }
122         }
123         
124         return errors;
125     }
