46  public class JavaDocParserTest {
47      private static final JDToken ONEEOL = _JDEOL("\n");
48      private static final JDToken ONEWS = _JDWhiteSpace(" ");
49      private static final List<JDToken> NO_TOKENS = Util.<JDToken>list();
50      private static final List<JDTagSection> NO_TAG_SECTIONS = Util.<JDTagSection>list();

52      @Test
53      public void testLookahead() {
54          // to get coverage of the null case in lookahead
55          final JavaDocParserImpl impl = new JavaDocParserImpl(new StringReader("foo1 foo2"));
56          impl.token = new Token(0, "bar1");
57          final Token token1 = impl.lookahead();
58          assertEquals("foo1", token1.image);
59          
60          // to get coverage of the non-null case in lookahead
61          impl.token = new Token(0, "bar");
62          impl.token.next = new Token(0, "bar2");
63          final Token token2 = impl.lookahead();
64          assertEquals("bar2", token2.image);      
65      }

67      @Test
68      public void testGeneralSection() {
69          test("/** */", _JavaDocComment("/**", list(ONEWS), NO_TAG_SECTIONS, "*/"));
70          test("/*** **/", _JavaDocComment("/***", list(ONEWS), NO_TAG_SECTIONS, "**/"));
71          test("/** * */", _JavaDocComment("/**", list(ONEWS, _JDAsterisk(), ONEWS), NO_TAG_SECTIONS, "*/"));
72          test("/** *\n */", _JavaDocComment("/**", list(ONEWS, _JDAsterisk(), ONEEOL, ONEWS), NO_TAG_SECTIONS, "*/"));
73          test("/**\n * hello\n * world\n */", _JavaDocComment("/**", list(ONEEOL, ONEWS, _JDAsterisk(), ONEWS, _JDWord("hello"), ONEEOL, ONEWS, _JDAsterisk(), ONEWS, _JDWord("world"), ONEEOL, ONEWS), NO_TAG_SECTIONS, "*/"));
74          test("/**\n * hello @foo */", _JavaDocComment("/**", list(ONEEOL, ONEWS, _JDAsterisk(), ONEWS, _JDWord("hello"), ONEWS, _JDTag("@foo"), ONEWS), NO_TAG_SECTIONS, "*/"));
75          test("/**\n * hello\n * * @world\n */", _JavaDocComment("/**", list(ONEEOL, ONEWS, _JDAsterisk(), ONEWS, _JDWord("hello"), ONEEOL, ONEWS, _JDAsterisk(), ONEWS, _JDAsterisk(), ONEWS, _JDTag("@world"), ONEEOL, ONEWS), NO_TAG_SECTIONS, "*/"));
76          test("/** /** */", _JavaDocComment("/**", list(ONEWS, _JDWord("/**"), ONEWS), NO_TAG_SECTIONS, "*/"));
77      }

79      @Test
80      public void testTagSections() {
81          test("/**@Foo*/", _JavaDocComment("/**", NO_TOKENS, list(_JDTagSection("@Foo", list(_JDTag("@Foo")))), "*/"));        
82          test("/**@Foo hello\n * world*/", _JavaDocComment("/**", NO_TOKENS, list(_JDTagSection("@Foo", list(_JDTag("@Foo"), ONEWS, _JDWord("hello"), ONEEOL, ONEWS, _JDAsterisk(), ONEWS, _JDWord("world")))), "*/"));        
83          test("/**@Foo hello\n * world\n@Bar whatever*/", _JavaDocComment("/**", NO_TOKENS, list(_JDTagSection("@Foo", list(_JDTag("@Foo"), ONEWS, _JDWord("hello"), ONEEOL, ONEWS, _JDAsterisk(), ONEWS, _JDWord("world"), ONEEOL)), _JDTagSection("@Bar", list(_JDTag("@Bar"), ONEWS, _JDWord("whatever")))), "*/"));        
84          test("/** @hello\n * */\n", _JavaDocComment("/**", list(ONEWS), list(_JDTagSection("@hello", list(_JDTag("@hello"), ONEEOL, ONEWS, _JDAsterisk(), ONEWS))), "*/"));
85      }
