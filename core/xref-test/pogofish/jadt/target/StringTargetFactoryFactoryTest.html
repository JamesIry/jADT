18  import static org.junit.Assert.*;
19  
20  import java.util.List;
21  import java.util.Map;
22  
23  import org.junit.Test;
24  
25  /**
26   * Test that the factory factory actually factories
27   *
28   * @author jiry
29   */
30  public class StringTargetFactoryFactoryTest {
31      /**
32       * Make sure the results map is correct
33       */
34      @Test
35      public void test() {
36          final StringTargetFactoryFactory ugh = new StringTargetFactoryFactory();
37          
38          final StringTargetFactory sf1 = ugh.createTargetFactory("baseDir1");
39          final StringTargetFactory sf2 = ugh.createTargetFactory("baseDir1");
40          final StringTargetFactory sf3 = ugh.createTargetFactory("baseDir2");
41          
42          assertNotSame(sf1, sf2);
43          assertNotSame(sf2, sf3);
44          assertNotSame(sf1, sf3);
45          
46          final Map<String, List<StringTargetFactory>> results = ugh.results();
47          assertEquals(2, results.size());
48          
49          final List<StringTargetFactory> result1 = results.get("baseDir1");
50          assertEquals(2, result1.size());
51          assertSame(sf1, result1.get(0));
52          assertSame(sf2, result1.get(1));
53          
54          final List<StringTargetFactory> result2 = results.get("baseDir2");
55          assertEquals(1, result2.size());
56          assertSame(sf3, result2.get(0));
57      }
