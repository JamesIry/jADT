49      /* (non-Javadoc)
50       * @see com.pogofish.jadt.emitter.ClassBodyEmitter#constructorFactory(com.pogofish.jadt.emitter.Sink, java.lang.String, java.lang.String, com.pogofish.jadt.ast.Constructor)
51       */
52      @Override
53      public void constructorFactory(Sink sink, String dataTypeName, String factoryName, List<String> typeParametrs, Constructor constructor) {
54          if (constructor.args.isEmpty()) {
55          	if (typeParametrs.isEmpty()) {
56              	logger.finest("Generating no args, no types factory for " + constructor.name);
57          	} else {
58              	logger.finest("Generating no args, type arguments factory for " + constructor.name);
59          		sink.write("   @SuppressWarnings(\"rawtypes\")\n");
60          	}
61              sink.write("   private static final " + dataTypeName + " _" + factoryName + " = new " + constructor.name + "();\n");
62              sink.write(ASTPrinter.printComments("   ", commentProcessor.leftAlign(commentProcessor.javaDocOnly(constructor.comments))));
63          	if (!typeParametrs.isEmpty()) {
64          		sink.write("   @SuppressWarnings(\"unchecked\")\n");
65          	}
66          	sink.write("   public static final ");
67          	emitParameterizedTypeName(sink, typeParametrs);
68          	sink.write(" ");
69          	sink.write(dataTypeName);
70          	emitParameterizedTypeName(sink, typeParametrs);
71          	sink.write(" _" + factoryName + "() { return _" + factoryName + "; }");
72          } else {
73          	logger.finest("Generating args factory for " + constructor.name);
74              sink.write(ASTPrinter.printComments("   ", commentProcessor.leftAlign(constructor.comments)));
75              sink.write("   public static final ");
76              emitParameterizedTypeName(sink, typeParametrs);
77          	sink.write(" ");
78          	sink.write(dataTypeName);
79              emitParameterizedTypeName(sink, typeParametrs);
80              sink.write(" _" + factoryName + "(");
81              constructorArgs(sink, constructor, true);
82              sink.write(") { return new " + constructor.name);
83              emitParameterizedTypeName(sink, typeParametrs);
84              sink.write("("); 
85              constructorArgs(sink, constructor, false);
86              sink.write("); }");            
87          }
88      }

90      private void constructorArgs(Sink sink, Constructor constructor, boolean withTypes) {
91          boolean first = true;
92          for (Arg arg : constructor.args) {
93              if (first) {
94                  first = false;
95              } else {
96                  sink.write(", ");
97              }
98              sink.write(constructorArg(arg, withTypes));
99          }
100     }

103     /* (non-Javadoc)
104      * @see com.pogofish.jadt.emitter.ClassBodyEmitter#emitConstructorMethod(com.pogofish.jadt.emitter.Sink, com.pogofish.jadt.ast.Constructor)
105      */
106     @Override
107     public void emitConstructorMethod(Sink sink, String indent, Constructor constructor) {
108     	logger.finest("Generating constructor method for " + constructor.name);
109     	final List<JavaComment> javaDoc = commentProcessor.leftAlign(commentProcessor.javaDocOnly(constructor.comments));
110     	for (Arg arg : constructor.args) {
111     	    final List<JavaComment> paramDoc = commentProcessor.paramDoc(arg.name, javaDoc);
112     	    sink.write(ASTPrinter.printComments(indent, paramDoc));
113             final String finalName = arg.modifiers.contains(ArgModifier._Final()) ? "final " : "";
114             sink.write(indent + "public " + finalName + ASTPrinter.print(arg.type) + " " + arg.name + ";\n");
115         }
116     	sink.write("\n");
117         sink.write(ASTPrinter.printComments(indent, commentProcessor.leftAlign(commentProcessor.stripTags(CONSTRUCTOR_METHOD_STRIP, commentProcessor.javaDocOnly(constructor.comments)))));    	
118         sink.write(indent + "public " + constructor.name + "("); 
119         constructorArgs(sink, constructor, true);        
120         sink.write(") {");
121         for (Arg arg : constructor.args) {
122             sink.write("\n" + indent + "   this." + arg.name + " = " + arg.name + ";");
123         }
124         sink.write("\n" + indent + "}");
125     }

133         sink.write(indent + "@Override\n");
134         sink.write(indent + "public String toString() {\n");
135         sink.write(indent + "   return \"" + constructor.name);
136         if (!constructor.args.isEmpty()) {
137             sink.write("(");
138             boolean first = true;
139             for (Arg arg : constructor.args) {
140                 if (first) {
141                     first = false;
142                 } else {
143                     sink.write(", ");
144                 }
145                 sink.write(arg.name + " = \" + " + arg.name + " + \"");
146             }
147             sink.write(")");
148         }
149         sink.write("\";\n");
150         sink.write(indent + "}");
151     }

153     /* (non-Javadoc)
154      * @see com.pogofish.jadt.emitter.ClassBodyEmitter#emitEquals(com.pogofish.jadt.emitter.Sink, com.pogofish.jadt.ast.Constructor)
155      */
156     @Override
157     public void emitEquals(final Sink sink, final String indent, Constructor constructor, List<String> typeArguments) {
158     	logger.finest("Generating equals() for " + constructor.name);
159         sink.write(indent + "@Override\n");
160         sink.write(indent + "public boolean equals(Object obj) {\n");
161         sink.write(indent + "   if (this == obj) return true;\n");
162         sink.write(indent + "   if (obj == null) return false;\n");
163         sink.write(indent + "   if (getClass() != obj.getClass()) return false;\n");
164         if (!constructor.args.isEmpty()) {
165         	if (!typeArguments.isEmpty()) {
166         		sink.write(indent + "   @SuppressWarnings(\"rawtypes\")");
167         	}
168             sink.write(indent + "   " + constructor.name + " other = (" + constructor.name + ")obj;\n");
169             
170             for (final Arg arg : constructor.args) {
171                 arg.type._switch(new Type.SwitchBlock(){
172                     @Override
173                     public void _case(Ref x) {
174                         x.type._switch(new RefType.SwitchBlock() {
175                             @Override
176                             public void _case(ClassType x) {
177                                 sink.write(indent + "   if (" + arg.name + " == null) {\n");
178                                 sink.write(indent + "      if (other." + arg.name + " != null) return false;\n");
179                                 sink.write(indent + "   } else if (!" + arg.name + ".equals(other." + arg.name + ")) return false;\n");
180                             }
181 
182                             @Override
183                             public void _case(ArrayType x) {
184                                 sink.write(indent + "   if (!java.util.Arrays.equals(" + arg.name + ", other." + arg.name + ")) return false;\n");
185                             }});
186                     }
187 
188                     @Override
189                     public void _case(Primitive x) {
190                         sink.write(indent + "   if (" + arg.name + " != other." + arg.name + ") return false;\n");
191                     }});
192             }
193         }
194         sink.write(indent + "   return true;\n");
195         sink.write(indent + "}");
196     }

198     /* (non-Javadoc)
199      * @see com.pogofish.jadt.emitter.ClassBodyEmitter#emitHashCode(com.pogofish.jadt.emitter.Sink, com.pogofish.jadt.ast.Constructor)
200      */
201     @Override
202     public void emitHashCode(final Sink sink, final String indent, Constructor constructor) {
203     	logger.finest("Generating hashCode() for " + constructor.name);
204         sink.write(indent + "@Override\n");
205         sink.write(indent + "public int hashCode() {\n");
206         if (constructor.args.isEmpty()) {
207             sink.write(indent + "    return 0;\n");
208         } else {
209             sink.write(indent + "    final int prime = 31;\n");
210             sink.write(indent + "    int result = 1;\n");
211             for (final Arg arg : constructor.args) {
212                 arg.type._switch(new Type.SwitchBlock(){
213                     @Override
214                     public void _case(Ref x) {
215                         x.type._switch(new RefType.SwitchBlock() {
216                             @Override
217                             public void _case(ClassType x) {
218                                 sink.write(indent + "    result = prime * result + ((" + arg.name + " == null) ? 0 : " + arg.name + ".hashCode());\n");                
219                             }
220 
221                             @Override
222                             public void _case(ArrayType x) {
223                                 sink.write(indent + "    result = prime * result + java.util.Arrays.hashCode(" + arg.name + ");\n");                
224                             }});
225                     }
226 
227                     @Override
228                     public void _case(Primitive x) {
229                         x.type._switch(new PrimitiveType.SwitchBlockWithDefault() {
230 
231                             @Override
232                             public void _case(BooleanType x) {
233                                 sink.write(indent + "    result = prime * result + (" + arg.name + " ? 1 : 0);\n");
234                             }
235 
236                             @Override
237                             public void _case(IntType x) {
238                                 uncastedHashChunk(sink, arg);
239                             }
240 
241                             private void uncastedHashChunk(final Sink sink, final Arg arg) {
242                                 sink.write(indent + "    result = prime * result + " + arg.name + ";\n");
243                             }                            
244                             
245                             @Override
246                             public void _case(ByteType x) {
247                                 uncastedHashChunk(sink, arg);
248                             }
249 
250                             @Override
251                             public void _case(CharType x) {
252                                 uncastedHashChunk(sink, arg);
253                             }
254 
255                             @Override
256                             public void _case(ShortType x) {
257                                 uncastedHashChunk(sink, arg);
258                             }
259 
260                             @Override
261                             public void _default(PrimitiveType x) {
262                                 sink.write(indent + "    result = prime * result + (int)" + arg.name + ";\n");
263                             }});
264                     }});
265             }
266             sink.write(indent + "    return result;\n");
267         }
268         sink.write(indent + "}");
269     }

273     @Override
274 	public void emitParameterizedTypeName(Sink sink, List<String> typeArguments) {
275 		if (!typeArguments.isEmpty()) {
276 			sink.write("<");
277 			boolean first = true;
278 			for (String typeArgument : typeArguments) {
279 				if (first) {
280 					first = false;
281 				} else {
282 					sink.write(", ");
283 				}
284 				sink.write(typeArgument);
285 			}
286 			sink.write(">");
287 		}

289 	}    
290 	private String constructorArg(Arg arg, boolean withType) {
