30  /**
31   * Make sure FileSourceFactory does its job properly
32   *
33   * @author jiry
34   */
35  public class FileSourceFactoryTest {
36      /**
37       * On a valid file should get the data from the file.  This test must write to the file system
38       * @throws IOException
39       */
40      @Test
41      public void testValidFile() throws IOException {
42          final File temp = File.createTempFile("testFactory", "java");
43          try {
44              final BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(temp), "UTF-8"));
45              try {
46                  writer.write("hello");
47              } finally {
48                  writer.close();
49              }
50              
51              final SourceFactory factory = new FileSourceFactory();
52              final Source source = factory.createSource(temp.getAbsolutePath());
53              try {
54                  final BufferedReader reader = new BufferedReader(source.getReader());
55                  assertEquals("hello", reader.readLine());
56                  assertEquals(temp.getAbsolutePath(), source.getSrcInfo());
57              } finally {
58                  source.close();
59              }
60          } finally {
61              temp.delete();
62          }
63      }
64  
65      /**
66       * This tst attempts to open a file that does not exist which should result in an exception.  It should not create any files
67       */
68      @Test
69      public void testInValidFile() throws IOException {
70          final File temp = File.createTempFile("testFactory", "java");
71          temp.delete();
72          
73          final SourceFactory factory = new FileSourceFactory();
74          try {
75              final Source source = factory.createSource(temp.getAbsolutePath());
76              source.close();
77              fail("Did not get an exception");
78          } catch (RuntimeException e) {
79              // yay
80          }
81      }
