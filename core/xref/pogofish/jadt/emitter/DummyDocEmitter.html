16  package pogofish.jadt.emitter;
17  
18  import pogofish.jadt.ast.Doc;
19  import pogofish.jadt.target.Target;
20  import pogofish.jadt.target.TargetFactory;
21  
22  /**
23   * Dummy emitter for Doc that just outputs the srcInfo
24   *
25   * @author jiry
26   */
27  public class DummyDocEmitter implements DocEmitter {
28      private final Doc testDoc;
29      private final String className;
30      
31      /**
32       * Takes a String className that will be used to create a target and a document that it will assert
33       * it gets during emit
34       */
35      public DummyDocEmitter(Doc testDoc, String className) {
36          super();
37          this.testDoc = testDoc;
38          this.className = className;
39      }
40  
41  
42      @Override
43      public void emit(TargetFactory factory, Doc doc) {
44          if (testDoc != doc) {
45              throw new RuntimeException("testDoc and doc argument were not the same");
46          }
47          final Target target = factory.createTarget(className);
48          try {
49              target.write(doc.srcInfo);
50          } finally {
51              target.close();
52          }
53      }
54  
55  }
