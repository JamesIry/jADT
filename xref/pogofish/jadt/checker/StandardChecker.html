16  package pogofish.jadt.checker;
17  
18  import java.util.HashSet;
19  import java.util.Set;
20  
21  import pogofish.jadt.ast.*;
22  
23  /**
24   * Standard checker that finds problems in Doc documents
25   *
26   * @author jiry
27   */
28  public class StandardChecker implements Checker {
29      
30      /**
31       * Checks a documents for duplicate dataType names and calls check(DataType) on each one
32       * 
33       * @param doc Doc to check
34       * @return Set<SemanticException> with the problems or empty set if there are none
35       */
36      @Override 
37      public Set<SemanticException> check(Doc doc) {
38          final Set<SemanticException> errors = new HashSet<SemanticException>();
39          final Set<String> dataTypeNames = new HashSet<String>();
40          for (DataType dataType : doc.dataTypes) {
41              errors.addAll(check(dataType));
42              if (dataTypeNames.contains(dataType.name)) {
43                  errors.add(new DuplicateDataTypeException(dataType.name));
44              } else {
45                  dataTypeNames.add(dataType.name);
46              }
47          }
48          return errors;
49      }
50  
51      /**
52       * Checks a data type for duplicate constructor names or constructors having the same name
53       * as the data type
54       * 
55       * @param dataType DataType to check
56       * @return Set<SemanticException> with the problems or empty set if there are none
57       */
58      private Set<SemanticException> check(DataType dataType) {
59          final Set<SemanticException> errors = new HashSet<SemanticException>();
60          final Set<String> constructorNames = new HashSet<String>();
61          if (dataType.constructors.size() > 1) {
62              for(Constructor constructor : dataType.constructors) {
63                  if (dataType.name.equals(constructor.name)) {
64                      errors.add(new ConstructorDataTypeConflictException(dataType.name, constructor.name));
65                  }
66                  if (constructorNames.contains(constructor.name)) {
67                      errors.add(new DuplicateConstructorException(dataType.name, constructor.name));
68                  } else {
69                      constructorNames.add(constructor.name);
70                  }
71              }
72          }
73          return errors;
74      }

76  }
