29  public class IOExceptionActionTest {
30      /**
31       * What happens when there's no exception?
32       */
33      @Test
34      public void testNoException() {
35          final String result = new IOExceptionAction<String>() {
36  
37              @Override
38              public String doAction() throws IOException {
39                  return "hello";
40              }
41          }.execute();
42  
43          assertEquals("hello", result);
44      }
45  
46      /**
47       * What happens when there is one?
48       */
49      @Test
50      public void testException() {
51          try {
52              final String result = new IOExceptionAction<String>() {
53  
54                  @Override
55                  public String doAction() throws IOException {
56                      throw new IOException("uh oh");
57                  }
58              }.execute();
59              fail("Execution did not cause an exception got " + result);
60          } catch (RuntimeException e) {
61              assertTrue("Contained exception was the wrong type", e.getCause() instanceof IOException);
62          }
63      }
